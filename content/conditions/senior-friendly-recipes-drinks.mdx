/* ===== groups for easy scanning ===== */
const GROUPS = [
  { id:'Meals',   badge:'Meals',   filter:(r)=> hasAny(r, ['break','lunch','din']) },
  { id:'Snacks & Sides', badge:'Snacks', filter:(r)=> hasAny(r, ['snack']) },
  { id:'Drinks & Hydration', badge:'Drinks', filter:(r)=> hasAny(r, ['drink']) },
  { id:'Special Diets', badge:'Diet Sets', filter:(r)=> hasAny(r, ['kidney','diab','heart','reflux','k-low','salt','soft','fiber','protein']) }
];

function hasAny(item, list){
  const cats = item.cats || [];
  return list.some(c => cats.includes(c));
}

function tagLabel(cat){
  switch(cat){
    case 'kidney': return 'Kidney-Safe';
    case 'heart': return 'Heart-Friendly';
    case 'diab': return 'Diabetes-Friendly';
    case 'reflux': return 'Reflux-Safe';
    case 'salt': return 'Low-Salt';
    case 'k-low': return 'Low-Potassium';
    case 'fiber': return 'High-Fiber';
    case 'protein': return 'Protein';
    case 'soft': return 'Soft/Easy-to-Chew';
    case 'break': return 'Breakfast';
    case 'lunch': return 'Lunch';
    case 'din': return 'Dinner';
    case 'snack': return 'Snack';
    case 'drink': return 'Drink';
    case 'veg': return 'Vegetarian';
    case 'fish': return 'Fish';
    case 'chicken': return 'Chicken';
    default: return 'General';
  }
}

function cardHTML(it){
  const cls = (it.cats && it.cats[0]) || 'veg';
  const tags = (it.cats||['veg']).map(c=>`<span class="tag">${tagLabel(c)}</span>`).join("");
  return `
    <a href="${it.href||'#'}" class="card ${cls}" data-cats="${(it.cats||[]).join(' ')}" data-tags="${it.tags||''}">
      <h3>${it.title}</h3>
      <p>${it.desc}</p>
      <div class="meta">${tags}</div>
    </a>
  `;
}

const q = document.getElementById('q');
const chips = Array.from(document.querySelectorAll('.chip'));
const groupsEl = document.getElementById('groups');
let activeChip = 'all';

function render(){
  const term = (q.value||"").trim().toLowerCase();
  const items = RECS.filter(it=>{
    const hay = (it.title + " " + it.desc + " " + (it.tags||"") + " " + (it.cats||[]).join(" ")).toLowerCase();
    const chipMatch = (activeChip==='all') ? true : (it.cats||[]).includes(activeChip);
    return chipMatch && (term ? hay.includes(term) : true);
  });

  const html = GROUPS.map(g=>{
    const subset = items.filter(g.filter);
    if(!subset.length) return "";
    return `
      <section class="group" aria-labelledby="${g.id}-head">
        <h2 id="${g.id}-head"><span class="badge">${g.badge}</span> ${g.id}</h2>
        <div class="grid">
          ${subset.map(cardHTML).join("")}
        </div>
      </section>
    `;
  }).join("");

  groupsEl.innerHTML = html || `<p style="text-align:center;color:#6b7280">No items match yet. Try a different word.</p>`;
}

/* events */
q && q.addEventListener('input', render);
q && q.addEventListener('keydown', (e)=>{
  if(e.key === 'Enter'){
    const first = document.querySelector('.card');
    if(first) first.focus();
  }
});
chips.forEach(btn=>{
  btn.addEventListener('click', ()=>{
    chips.forEach(b=>b.setAttribute('aria-pressed','false'));
    btn.setAttribute('aria-pressed','true');
    activeChip = btn.dataset.chip || 'all';
    render();
  });
});

/* initial render */
render();
