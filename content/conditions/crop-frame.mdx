<script>
(()=> {
  const $ = s => document.querySelector(s);

  // Elements
  const pv = $('#pv'), ctx = pv.getContext('2d');
  const wrap = $('#canvasWrap'), drop = $('#drop');
  const gridBox = $('#grid');

  const fileInput = $('#file'), cameraInput = $('#cameraFile');
  const aspectEl = $('#aspect'), custWEl = $('#custW'), custHEl = $('#custH'), orientEl = $('#orient');
  const outWEl = $('#outW'), outHEl = $('#outH'), formatEl = $('#format'), qualityEl = $('#quality'), zoomEl = $('#zoom');
  const rotL = $('#rotL'), rotR = $('#rotR'), flipHBtn = $('#flipH'), flipVBtn = $('#flipV');
  const presets = $('#presets'), downloadBtn = $('#downloadBtn'), resetBtn = $('#resetBtn');
  const meta = $('#meta'), frameInfo = $('#frameInfo'), exportInfo = $('#exportInfo'), estInfo = $('#estInfo');

  // State
  let img = null, imgURL = null, baseName = 'image';
  let zoom = 100, rot = 0, flipH = false, flipV = false;
  let offsetX = 0, offsetY = 0;
  let dragging = false, lastX = 0, lastY = 0;
  let pinch = null; // {d, zoom}
  let exportBlob = null;

  // Helpers
  const clamp = (v,min,max)=>Math.max(min,Math.min(max,v));
  function parseAspect(val){
    if(val==='free') return null;
    if(val==='original' && img){
      const r = (rot%180!==0) ? (img.height/img.width) : (img.width/img.height);
      return r; // width/height? we want w/h ratio
    }
    if(val==='original' && !img) return 1;
    if(val==='custom'){
      const w = +custWEl.value||1, h = +custHEl.value||1;
      return Math.max(0.0001, w/h);
    }
    const [w,h] = val.split(':').map(Number);
    return w/h;
  }
  function applyOrientationRatio(r){
    const o = orientEl.value;
    if(o==='auto') return r;
    if(o==='land') return r>=1 ? r : 1/r;
    if(o==='port') return r<=1 ? r : 1/r;
  }

  // Canvas sizing to container (for sharp preview)
  function fitCanvasToContainer(){
    const w = Math.max(320, wrap.clientWidth - 24);
    const h = Math.max(260, Math.round(w*0.66));
    pv.width = w; pv.height = h;
  }

  function getFrameRect(){
    const pad = 12;
    const W = pv.width - pad*2, H = pv.height - pad*2;
    const a = parseAspect(aspectEl.value);
    if(a==null){ // free = whole area
      return {x:pad, y:pad, w:W, h:H};
    }
    const ratio = applyOrientationRatio(a);
    let fw = W, fh = Math.round(W/ratio);
    if(fh>H){ fh = H; fw = Math.round(H*ratio); }
    const x = Math.round((pv.width - fw)/2);
    const y = Math.round((pv.height - fh)/2);
    return {x,y,w:fw,h:fh};
  }

  function drawGrid(rect){
    gridBox.style.display = 'block';
    gridBox.style.left = rect.x+'px';
    gridBox.style.top = rect.y+'px';
    gridBox.style.width = rect.w+'px';
    gridBox.style.height = rect.h+'px';

    // build thirds if not present
    if(!gridBox.querySelector('.gline')){
      for(let i=1;i<=2;i++){
        const v=document.createElement('div'); v.className='gline v';
        v.style.left = (i*100/3)+'%'; gridBox.appendChild(v);
        const h=document.createElement('div'); h.className='gline h';
        h.style.top  = (i*100/3)+'%'; gridBox.appendChild(h);
      }
    }
  }

  function getRotatedDims(w,h,rotDeg){
    return (rotDeg%180===0) ? {w,h} : {w:h, h:w};
  }

  function drawPreview(){
    fitCanvasToContainer();
    ctx.clearRect(0,0,pv.width,pv.height);

    const rect = getFrameRect();

    // Dark outside mask
    ctx.save();
    ctx.fillStyle = 'rgba(2,6,23,.55)';
    ctx.beginPath();
    ctx.rect(0,0,pv.width,pv.height);
    ctx.rect(rect.x, rect.y, rect.w, rect.h);
    ctx.fill('evenodd');
    ctx.restore();

    // Frame border
    ctx.save();
    ctx.lineWidth = 1;
    ctx.strokeStyle = 'rgba(255,255,255,.85)';
    ctx.strokeRect(rect.x+.5, rect.y+.5, rect.w-1, rect.h-1);
    ctx.restore();

    // Image
    if(img){
      const {w:iw0, h:ih0} = getRotatedDims(img.width, img.height, rot);
      const scaleBase = Math.max(rect.w/iw0, rect.h/ih0);
      const scale = scaleBase * (zoom/100);

      const cx = rect.x + rect.w/2 + offsetX;
      const cy = rect.y + rect.h/2 + offsetY;

      ctx.save();
      ctx.imageSmoothingEnabled = true;
      ctx.imageSmoothingQuality = 'high';
      ctx.translate(cx, cy);
      ctx.rotate(rot * Math.PI/180);
      ctx.scale(flipH?-1:1, flipV?-1:1);
      ctx.scale(scale, scale);
      ctx.drawImage(img, -img.width/2, -img.height/2);
      ctx.restore();

      frameInfo.textContent = `Frame: ${rect.w}×${rect.h}px (preview)`;
      exportInfo.textContent = `Export: ${outWEl.value}×${outHEl.value}px`;
      meta.hidden = false;
    } else {
      meta.hidden = true;
    }

    drawGrid(rect);
    // Prepare export blob estimate (debounced-ish)
    scheduleExportEstimate();
    downloadBtn.disabled = !img;
  }

  // Wheel zoom
  pv.addEventListener('wheel', (e)=>{
    e.preventDefault();
    const delta = e.deltaY>0 ? -5 : 5;
    zoom = clamp(zoom + delta, 10, 500);
    zoomEl.value = zoom;
    drawPreview();
  }, {passive:false});

  // Drag
  function startDrag(x,y){ dragging=true; lastX=x; lastY=y; }
  function moveDrag(x,y){
    if(!dragging) return;
    offsetX += (x - lastX);
    offsetY += (y - lastY);
    lastX = x; lastY = y;
    drawPreview();
  }
  function endDrag(){ dragging=false; }

  pv.addEventListener('mousedown', e=> startDrag(e.clientX, e.clientY));
  window.addEventListener('mousemove', e=> moveDrag(e.clientX, e.clientY));
  window.addEventListener('mouseup', endDrag);

  // Touch: drag + pinch
  pv.addEventListener('touchstart', e=>{
    if(e.touches.length===1){
      const t = e.touches[0]; startDrag(t.clientX, t.clientY);
    }else if(e.touches.length===2){
      dragging=false;
      const dx = e.touches[0].clientX - e.touches[1].clientX;
      const dy = e.touches[0].clientY - e.touches[1].clientY;
      pinch = { d: Math.hypot(dx,dy), zoom };
    }
  }, {passive:true});
  pv.addEventListener('touchmove', e=>{
    if(e.touches.length===1 && dragging){
      const t = e.touches[0]; moveDrag(t.clientX, t.clientY);
    }else if(e.touches.length===2 && pinch){
      const dx = e.touches[0].clientX - e.touches[1].clientX;
      const dy = e.touches[0].clientY - e.touches[1].clientY;
      const d = Math.hypot(dx,dy);
      const ratio = d / (pinch.d || d);
      zoom = clamp(Math.round((pinch.zoom || zoom) * ratio), 10, 500);
      zoomEl.value = zoom;
      drawPreview();
    }
  }, {passive:true});
  pv.addEventListener('touchend', ()=>{ dragging=false; pinch=null; }, {passive:true});

  // Aspect & outputs
  function updateOutputSizeFromAspect(){
    const a = parseAspect(aspectEl.value);
    if(a==null){ // free
      return; // keep manual sizes
    }
    const r = applyOrientationRatio(a);
    const w = +outWEl.value || 1080;
    outHEl.value = Math.max(1, Math.round(w / r));
  }

  aspectEl.addEventListener('change', ()=>{ updateOutputSizeFromAspect(); drawPreview(); });
  [custWEl, custHEl, orientEl].forEach(el=> el.addEventListener('input', ()=>{ if(aspectEl.value==='custom'){ updateOutputSizeFromAspect(); drawPreview(); }}));

  outWEl.addEventListener('input', ()=>{ updateOutputSizeFromAspect(); drawPreview(); });
  outHEl.addEventListener('input', ()=>{ drawPreview(); });

  // Zoom / rotate / flip
  zoomEl.addEventListener('input', ()=>{ zoom = +zoomEl.value || 100; drawPreview(); });
  rotL.addEventListener('click', ()=>{ rot = (rot - 90 + 360) % 360; drawPreview(); });
  rotR.addEventListener('click', ()=>{ rot = (rot + 90) % 360; drawPreview(); });
  flipHBtn.addEventListener('click', ()=>{ flipH = !flipH; drawPreview(); });
  flipVBtn.addEventListener('click', ()=>{ flipV = !flipV; drawPreview(); });

  // Preset chips
  presets.addEventListener('click', e=>{
    const b = e.target.closest('.chip'); if(!b) return;
    aspectEl.value = b.dataset.a || 'free';
    outWEl.value = b.dataset.w || outWEl.value;
    outHEl.value = b.dataset.h || outHEl.value;
    updateOutputSizeFromAspect();
    drawPreview();
  });

  // File handling
  function loadFile(file){
    if(!file || !file.type?.startsWith('image/')) return;
    baseName = (file.name||'image').replace(/\.[^.]+$/,'');
    if(imgURL) URL.revokeObjectURL(imgURL);
    imgURL = URL.createObjectURL(file);
    const im = new Image();
    im.onload = ()=>{ img = im; fileInput.value=''; cameraInput.value=''; resetView(false); drawPreview(); };
    im.onerror = ()=>{ URL.revokeObjectURL(imgURL); fileInput.value=''; cameraInput.value=''; alert('Could not load this image.'); };
    im.src = imgURL;
  }

  // Drag & drop
  ['dragenter','dragover'].forEach(ev=>{
    drop.addEventListener(ev, e=>{ e.preventDefault(); drop.classList.add('drag'); });
  });
  ['dragleave','drop'].forEach(ev=>{
    drop.addEventListener(ev, e=>{ e.preventDefault(); drop.classList.remove('drag'); });
  });
  drop.addEventListener('drop', e=>{
    const f = e.dataTransfer?.files?.[0];
    if(f) loadFile(f);
  });

  // Real inputs (mobile-safe)
  fileInput.addEventListener('change', e=>{
    const f=e.target.files && e.target.files[0];
    if(f) loadFile(f);
  });
  cameraInput.addEventListener('change', e=>{
    const f=e.target.files && e.target.files[0];
    if(f) loadFile(f);
  });

  // Paste
  window.addEventListener('paste', e=>{
    const items = e.clipboardData?.items;
    if(!items) return;
    for(const it of items){
      if(it.type?.startsWith('image/')){ loadFile(it.getAsFile()); break; }
    }
  });

  // Export (build offscreen canvas with same transform, scaled to output)
  function exportImage(callback){
    if(!img) return;
    const rect = getFrameRect();
    const W = +outWEl.value||1080, H = +outHEl.value||1080;

    const off = document.createElement('canvas');
    off.width = clamp(Math.round(W), 1, 10000);
    off.height = clamp(Math.round(H), 1, 10000);
    const octx = off.getContext('2d');

    const {w:iw0, h:ih0} = getRotatedDims(img.width, img.height, rot);
    const scaleBase = Math.max(rect.w/iw0, rect.h/ih0);
    const scale = scaleBase * (zoom/100);

    // Map preview frame -> export canvas
    const s = off.width / rect.w; // scale factor from frame pixels to export pixels
    const cx = (rect.w/2 + offsetX) * s;
    const cy = (rect.h/2 + offsetY) * s;

    octx.save();
    octx.imageSmoothingEnabled = true;
    octx.imageSmoothingQuality = 'high';
    octx.translate(cx, cy);
    octx.rotate(rot*Math.PI/180);
    octx.scale(flipH?-1:1, flipV?-1:1);
    octx.scale(scale*s, scale*s);
    octx.drawImage(img, -img.width/2, -img.height/2);
    octx.restore();

    const mime = formatEl.value;
    const q = (mime==='image/png') ? 1 : (+qualityEl.value||92)/100;
    off.toBlob(b=> callback(b), mime, q);
  }

  let estTimer = null;
  function scheduleExportEstimate(){
    if(!img) { estInfo.textContent='Estimated file size: —'; return; }
    if(estTimer) cancelAnimationFrame(estTimer);
    estTimer = requestAnimationFrame(()=>{
      exportImage(blob=>{
        exportBlob = blob;
        estInfo.textContent = 'Estimated file size: ' + humanBytes(blob?.size||0);
      });
    });
  }
  function humanBytes(bytes){
    if(!bytes) return '—';
    const u=['B','KB','MB','GB']; let i=0; let v=bytes;
    while(v>1024 && i<u.length-1){ v/=1024; i++; }
    return v.toFixed(v<10?2:1)+' '+u[i];
  }

  downloadBtn.addEventListener('click', ()=>{
    if(!img) return;
    exportImage(blob=>{
      if(!blob) return;
      const extMap = {'image/jpeg':'jpg','image/png':'png','image/webp':'webp'};
      const ext = extMap[formatEl.value]||'jpg';
      const a = document.createElement('a');
      a.href = URL.createObjectURL(blob);
      a.download = `${baseName}-cropped.${ext}`;
      document.body.appendChild(a);
      a.click();
      setTimeout(()=>{ URL.revokeObjectURL(a.href); a.remove(); }, 500);
    });
  });

  function resetView(resetAspect=true){
    zoom=100; rot=0; flipH=false; flipV=false; offsetX=0; offsetY=0; zoomEl.value=100;
    if(resetAspect){
      aspectEl.value='free'; custWEl.value=''; custHEl.value='';
      outWEl.value=1080; outHEl.value=1080;
    }
  }
  resetBtn.addEventListener('click', ()=>{
    img=null; if(imgURL){ URL.revokeObjectURL(imgURL); imgURL=null; }
    baseName='image'; resetView(); drawPreview(); exportBlob=null; meta.hidden=true;
  });

  // Init
  window.addEventListener('resize', drawPreview);
  document.getElementById('vt-year').textContent = new Date().getFullYear();
  drawPreview();
})();
