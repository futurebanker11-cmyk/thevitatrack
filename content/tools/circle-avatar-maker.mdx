(()=> {
  const $ = s => document.querySelector(s);

  // Elements
  const pv = $('#pv'), ctx = pv.getContext('2d');
  const wrap = $('#canvasWrap'), drop = $('#drop');

  const fileInput = $('#file'), cameraInput = $('#cameraFile');
  const sizePxEl = $('#sizePx'), sizePresets = $('#sizePresets');
  const zoomEl = $('#zoom');
  const bgModeEl = $('#bgMode'), bgColorWrap = $('#bgColorWrap'), bgColorEl = $('#bgColor');
  const ringWEl = $('#ringW'), ringColorEl = $('#ringColor'), shadowEl = $('#shadow');
  const rotL = $('#rotL'), rotR = $('#rotR');
  const fmtEl = $('#format'), qEl = $('#quality');
  const downloadBtn = $('#downloadBtn'), resetBtn = $('#resetBtn');

  const meta = $('#meta'), origInfo = $('#origInfo'), expInfo = $('#expInfo'), estInfo = $('#estInfo');

  // State
  let img = null, imgURL = null, baseName = 'avatar';
  let zoom = 120;
  let offsetX = 0, offsetY = 0;
  let dragging = false, lastX = 0, lastY = 0, pinch = null;
  let rot = 0; // 0/90/180/270
  let lastFrameW = 1; // preview frame square width for export mapping
  let exportBlob = null;

  // Helpers
  const clamp = (v,min,max)=>Math.max(min,Math.min(max,v));
  function fitCanvas(){
    const w = Math.max(320, wrap.clientWidth - 24);
    const h = Math.max(320, Math.round(w*0.66));
    pv.width = w; pv.height = h;
  }
  function getRotDims(w,h,r){ return (r%180===0) ? {w,h} : {w:h, h:w}; }
  function humanBytes(bytes){
    if(!bytes) return '—';
    const u=['B','KB','MB','GB']; let i=0; let v=bytes;
    while(v>1024 && i<u.length-1){ v/=1024; i++; }
    return v.toFixed(v<10?2:1)+' '+u[i];
  }

  function getFrameRect(){
    // Square frame centered in preview (with padding)
    const pad = 18;
    const W = pv.width - pad*2, H = pv.height - pad*2;
    const side = Math.min(W, H);
    const x = Math.round((pv.width - side)/2);
    const y = Math.round((pv.height - side)/2);
    lastFrameW = side;
    return {x,y,w:side,h:side, r: side/2, cx: x+side/2, cy: y+side/2};
  }

  // Draw preview
  function draw(){
    fitCanvas();
    ctx.clearRect(0,0,pv.width,pv.height);
    const frame = getFrameRect();

    // Optional solid background inside circle (for preview)
    const showSolid = (bgModeEl.value==='solid');
    if(showSolid){
      ctx.save();
      ctx.beginPath(); ctx.arc(frame.cx, frame.cy, frame.r, 0, Math.PI*2);
      ctx.fillStyle = bgColorEl.value; ctx.fill();
      ctx.restore();
    }

    if(img){
      // Compute scale to cover the frame
      const {w:iw0, h:ih0} = getRotDims(img.width, img.height, rot);
      const baseScale = Math.max(frame.w/iw0, frame.h/ih0);
      const scale = baseScale * (zoom/100);

      // Clip to circle
      ctx.save();
      ctx.beginPath(); ctx.arc(frame.cx, frame.cy, frame.r, 0, Math.PI*2); ctx.clip();

      // Draw image with transforms
      ctx.imageSmoothingEnabled = true; ctx.imageSmoothingQuality = 'high';
      ctx.translate(frame.cx + offsetX, frame.cy + offsetY);
      ctx.rotate(rot*Math.PI/180);
      ctx.scale(scale, scale);
      ctx.drawImage(img, -img.width/2, -img.height/2);
      ctx.restore();
    }

    // Shadow (draw a filled circle to cast shadow)
    const sh = shadowEl.value;
    if(sh!=='none'){
      ctx.save();
      if(sh==='soft'){ ctx.shadowColor='rgba(0,0,0,.35)'; ctx.shadowBlur=14; }
      else if(sh==='strong'){ ctx.shadowColor='rgba(0,0,0,.55)'; ctx.shadowBlur=24; }
      ctx.beginPath(); ctx.arc(frame.cx, frame.cy, frame.r-1, 0, Math.PI*2);
      ctx.strokeStyle='rgba(0,0,0,0)'; ctx.lineWidth=0.001; // minimal stroke to trigger shadow
      ctx.stroke();
      ctx.restore();
    }

    // Ring (border) — keep fully inside circle
    const ringWExport = +ringWEl.value||0;
    if(ringWExport>0){
      const exportSize = +sizePxEl.value||512;
      const ringWPreview = ringWExport * (frame.w / exportSize);
      ctx.save();
      ctx.beginPath();
      ctx.arc(frame.cx, frame.cy, Math.max(0, frame.r - ringWPreview/2), 0, Math.PI*2);
      ctx.strokeStyle = ringColorEl.value;
      ctx.lineWidth = ringWPreview;
      ctx.stroke();
      ctx.restore();
    }

    if(img){
      origInfo.textContent = `Original: ${img.width}×${img.height}px`;
      expInfo.textContent  = `Export: ${+sizePxEl.value||512}×${+sizePxEl.value||512}px`;
      meta.hidden = false;
      scheduleEstimate();
      downloadBtn.disabled = false;
    }else{
      meta.hidden = true;
      downloadBtn.disabled = true;
    }
  }

  // Wheel zoom
  pv.addEventListener('wheel', (e)=>{
    e.preventDefault();
    const delta = e.deltaY>0 ? -5 : 5;
    zoom = clamp(zoom + delta, 10, 500);
    zoomEl.value = zoom;
    draw();
  }, {passive:false});

  // Drag & touch
  function startDrag(x,y){ dragging=true; lastX=x; lastY=y; }
  function moveDrag(x,y){
    if(!dragging) return;
    offsetX += (x - lastX);
    offsetY += (y - lastY);
    lastX = x; lastY = y;
    draw();
  }
  function endDrag(){ dragging=false; }

  pv.addEventListener('mousedown', e=> startDrag(e.clientX, e.clientY));
  window.addEventListener('mousemove', e=> moveDrag(e.clientX, e.clientY));
  window.addEventListener('mouseup', endDrag);

  pv.addEventListener('touchstart', e=>{
    if(e.touches.length===1){
      const t = e.touches[0]; startDrag(t.clientX, t.clientY);
    }else if(e.touches.length===2){
      dragging=false;
      const dx = e.touches[0].clientX - e.touches[1].clientX;
      const dy = e.touches[0].clientY - e.touches[1].clientY;
      pinch = { d: Math.hypot(dx,dy), zoom };
    }
  }, {passive:true});
  pv.addEventListener('touchmove', e=>{
    if(e.touches.length===1 && dragging){
      const t = e.touches[0]; moveDrag(t.clientX, t.clientY);
    }else if(e.touches.length===2 && pinch){
      const dx = e.touches[0].clientX - e.touches[1].clientX;
      const dy = e.touches[0].clientY - e.touches[1].clientY;
      const d = Math.hypot(dx,dy);
      const ratio = d / (pinch.d || d);
      zoom = clamp(Math.round((pinch.zoom || zoom) * ratio), 10, 500);
      zoomEl.value = zoom;
      draw();
    }
  }, {passive:true});
  pv.addEventListener('touchend', ()=>{ dragging=false; pinch=null; }, {passive:true});

  // Export
  function exportImage(cb){
    if(!img){ cb(null); return; }
    const size = clamp(+sizePxEl.value||512, 64, 8192);
    const off = document.createElement('canvas');
    off.width = size; off.height = size;
    const octx = off.getContext('2d');
    octx.imageSmoothingEnabled = true; octx.imageSmoothingQuality = 'high';
    const r = size/2, cx = size/2, cy = size/2;

    // Background fill for JPG, or if solid color chosen
    const mime = fmtEl.value;
    const wantTransparent = (bgModeEl.value==='transparent') && (mime!=='image/jpeg');
    if(!wantTransparent){
      octx.fillStyle = (bgModeEl.value==='solid') ? bgColorEl.value : '#ffffff';
      octx.fillRect(0,0,size,size);
    }

    // Clip to circle
    octx.save();
    octx.beginPath(); octx.arc(cx, cy, r, 0, Math.PI*2); octx.clip();

    // Map preview offsets to export
    const s = size / (lastFrameW || size); // preview->export scale
    const {w:iw0, h:ih0} = getRotDims(img.width, img.height, rot);
    const baseScale = Math.max(size/iw0, size/ih0);
    const scale = baseScale * (zoom/100);

    octx.translate(cx + offsetX*s, cy + offsetY*s);
    octx.rotate(rot*Math.PI/180);
    octx.scale(scale, scale);
    octx.drawImage(img, -img.width/2, -img.height/2);
    octx.restore();

    // Shadow
    const sh = shadowEl.value;
    if(sh!=='none'){
      octx.save();
      if(sh==='soft'){ octx.shadowColor='rgba(0,0,0,.35)'; octx.shadowBlur = Math.max(8, size*0.02); }
      else if(sh==='strong'){ octx.shadowColor='rgba(0,0,0,.55)'; octx.shadowBlur = Math.max(16, size*0.04); }
      octx.beginPath(); octx.arc(cx, cy, r-1, 0, Math.PI*2);
      octx.strokeStyle='rgba(0,0,0,0)'; octx.lineWidth=0.001; octx.stroke();
      octx.restore();
    }

    // Ring (inside edge)
    const ringW = +ringWEl.value||0;
    if(ringW>0){
      octx.save();
      octx.beginPath();
      octx.arc(cx, cy, Math.max(0, r - ringW/2), 0, Math.PI*2);
      octx.strokeStyle = ringColorEl.value;
      octx.lineWidth = ringW;
      octx.stroke();
      octx.restore();
    }

    const q = (mime==='image/png') ? 1 : (+qEl.value||92)/100;
    off.toBlob(b=> cb(b), mime, q);
  }

  let estRAF = null;
  function scheduleEstimate(){
    if(estRAF) cancelAnimationFrame(estRAF);
    estRAF = requestAnimationFrame(()=>{
      exportImage(b=>{
        exportBlob = b;
        estInfo.textContent = 'Estimated file size: ' + humanBytes(b?.size||0);
      });
    });
  }

  // File handling
  function loadFile(file){
    if(!file || !file.type?.startsWith('image/')) return;
    baseName = (file.name||'avatar').replace(/\.[^.]+$/,'');
    if(imgURL) URL.revokeObjectURL(imgURL);
    imgURL = URL.createObjectURL(file);
    const im = new Image();
    im.onload = ()=>{ img = im; fileInput.value=''; cameraInput.value=''; resetView(false); draw(); };
    im.onerror = ()=>{ URL.revokeObjectURL(imgURL); fileInput.value=''; cameraInput.value=''; alert('Could not load this image.'); };
    im.src = imgURL;
  }

  // Drag & drop
  ['dragenter','dragover'].forEach(ev=>{
    drop.addEventListener(ev, e=>{ e.preventDefault(); drop.classList.add('drag'); });
  });
  ['dragleave','drop'].forEach(ev=>{
    drop.addEventListener(ev, e=>{ e.preventDefault(); drop.classList.remove('drag'); });
  });
  drop.addEventListener('drop', e=>{
    const f = e.dataTransfer?.files?.[0];
    if(f) loadFile(f);
  });

  // Real inputs (mobile-safe)
  fileInput.addEventListener('change', e=>{
    const f=e.target.files && e.target.files[0];
    if(f) loadFile(f);
  });
  cameraInput.addEventListener('change', e=>{
    const f=e.target.files && e.target.files[0];
    if(f) loadFile(f);
  });

  // Paste
  window.addEventListener('paste', e=>{
    const items = e.clipboardData?.items;
    if(!items) return;
    for(const it of items){
      if(it.type?.startsWith('image/')){ loadFile(it.getAsFile()); break; }
    }
  });

  // Controls
  sizePresets.addEventListener('click', e=>{
    const b = e.target.closest('.chip'); if(!b) return;
    sizePxEl.value = b.dataset.s;
    draw();
  });
  zoomEl.addEventListener('input', ()=>{ zoom = +zoomEl.value||100; draw(); });

  bgModeEl.addEventListener('change', ()=>{
    bgColorWrap.classList.toggle('hide', bgModeEl.value!=='solid');
    draw();
  });
  bgColorEl.addEventListener('input', draw);

  ringWEl.addEventListener('input', draw);
  ringColorEl.addEventListener('input', draw);
  shadowEl.addEventListener('change', draw);

  rotL.addEventListener('click', ()=>{ rot = (rot - 90 + 360) % 360; draw(); });
  rotR.addEventListener('click', ()=>{ rot = (rot + 90) % 360; draw(); });

  [sizePxEl, fmtEl, qEl].forEach(el=>{
    el.addEventListener('input', draw);
    el.addEventListener('change', draw);
  });

  // Download
  downloadBtn.addEventListener('click', ()=>{
    if(!img) return;
    exportImage(blob=>{
      if(!blob) return;
      const extMap = {'image/jpeg':'jpg','image/png':'png','image/webp':'webp'};
      const ext = extMap[fmtEl.value]||'png';
      const a = document.createElement('a');
      a.href = URL.createObjectURL(blob);
      a.download = `${baseName}-avatar.${ext}`;
      document.body.appendChild(a);
      a.click();
      setTimeout(()=>{ URL.revokeObjectURL(a.href); a.remove(); }, 500);
    });
  });

  function resetView(resetAll=true){
    zoom = 120; offsetX = 0; offsetY = 0; rot = 0; zoomEl.value = 120;
    if(resetAll){
      sizePxEl.value = 512; bgModeEl.value='transparent'; bgColorEl.value='#ffffff'; bgColorWrap.classList.add('hide');
      ringWEl.value=8; ringColorEl.value='#ffffff'; shadowEl.value='soft';
      fmtEl.value='image/png'; qEl.value=92;
    }
  }
  resetBtn.addEventListener('click', ()=>{
    img=null; if(imgURL){ URL.revokeObjectURL(imgURL); imgURL=null; }
    baseName='avatar'; resetView(true); ctx.clearRect(0,0,pv.width,pv.height);
    exportBlob=null; meta.hidden=true; downloadBtn.disabled=true;
  });

  // Init
  window.addEventListener('resize', draw);
  document.getElementById('vt-year').textContent = new Date().getFullYear();
  draw();
})();
