  // Helpers
  function fitCanvas(){
    const w = Math.max(320, wrap.clientWidth - 24);
    const h = Math.max(260, Math.round(w*0.66));
    pv.width = w; pv.height = h;
  }
  function getRotDims(w,h,r){ return (r%180===0) ? {w,h} : {w:h, h:w}; }
  function humanBytes(bytes){
    if(!bytes) return '—';
    const u=['B','KB','MB','GB']; let i=0; let v=bytes;
    while(v>1024 && i<u.length-1){ v/=1024; i++; }
    return v.toFixed(v<10?2:1)+' '+u[i];
  }
  function clamp(v,min,max){ return Math.max(min,Math.min(max,v)); }

  function setStrengthHint(){
    if(modeEl.value==='blur'){
      $('#strengthHint').textContent = 'Blur radius (px)';
      strengthEl.min=2; strengthEl.max=40;
    }else{
      $('#strengthHint').textContent = 'Pixel size (px)';
      strengthEl.min=4; strengthEl.max=60;
    }
  }

  // Mapping preview coords -> image coords
  function getPreviewScale(){
    const {w:rw, h:rh} = getRotDims(srcC.width, srcC.height, 0);
    return Math.min(pv.width/rw, pv.height/rh);
  }
  function previewToImage(clientX, clientY){
    const rect = pv.getBoundingClientRect();
    const x = clientX - rect.left;
    const y = clientY - rect.top;
    const scale = getPreviewScale();
    const {w:rw, h:rh} = getRotDims(srcC.width, srcC.height, 0);
    const drawW = rw*scale, drawH = rh*scale;
    const dx = (pv.width - drawW)/2;
    const dy = (pv.height - drawH)/2;
    return { ix:(x - dx)/scale, iy:(y - dy)/scale };
  }

  function resetOffscreens(w,h){
    [srcC, filtC, maskC, blendC].forEach(c=>{ c.width=w; c.height=h; });

  }

  function rebuildFiltered(){
    filtX.save();
    if(modeEl.value==='blur'){
      filtX.clearRect(0,0,filtC.width,filtC.height);

      filtX.imageSmoothingEnabled = false;
      filtX.drawImage(tmp, 0,0, w,h, 0,0, filtC.width, filtC.height);
      filtX.imageSmoothingEnabled = true;
    }
    filtX.restore();
  }

    const feather = +featherEl.value||0;
    let maskToUse = maskC;
    if(feather>0){
      const m2 = document.createElement('canvas');
      m2.width = maskC.width; m2.height = maskC.height;
      const m2x = m2.getContext('2d');
      m2x.filter = `blur(${feather}px)`;
      m2x.drawImage(maskC,0,0);
      m2x.filter = 'none';
      maskToUse = m2;
    }

    // Create blended (filtered clipped by mask)
    blendX.clearRect(0,0,blendC.width,blendC.height);
    blendX.globalCompositeOperation = 'source-over';
    blendX.drawImage(filtC,0,0);
    blendX.globalCompositeOperation = 'destination-in';
    blendX.drawImage(maskToUse,0,0);
    blendX.globalCompositeOperation = 'source-over';

    // Draw base
    const scale = getPreviewScale();
    const dw = srcC.width*scale, dh = srcC.height*scale;
    const dx = (pv.width - dw)/2, dy = (pv.height - dh)/2;
    ctx.imageSmoothingEnabled = true; ctx.imageSmoothingQuality = 'high';
    ctx.drawImage(srcC, dx,dy, dw,dh);

    // Draw blended on top
    ctx.drawImage(blendC, dx,dy, dw,dh);

    // Optional mask overlay
    if(showMaskEl.checked){
      ctx.save();
      ctx.globalAlpha = 0.25;
      ctx.drawImage(maskC, dx,dy, dw,dh);
      ctx.restore();
    }

    // Live selection outline (rect/ellipse)
    if(selLive){
      ctx.save();
      ctx.strokeStyle = 'rgba(255,255,255,.9)';
      ctx.lineWidth = 2;
      ctx.setLineDash([6,6]);
      const x = dx + selLive.x*scale, y = dy + selLive.y*scale;
      const w = selLive.w*scale, h = selLive.h*scale;
      if(toolEl.value==='ellipse'){
        ctx.beginPath();
        ctx.ellipse(x+w/2, y+h/2, Math.abs(w/2), Math.abs(h/2), 0, 0, Math.PI*2);
        ctx.stroke();
      }else{ // rect
        ctx.strokeRect(x,y,w,h);
      }
      ctx.restore();
    }

    // Meta
    meta.hidden=false; downloadBtn.disabled=false;
    origInfo.textContent = `Original: ${srcC.width}×${srcC.height}px`;
    expInfo.textContent = `Export: ${srcC.width}×${srcC.height}px`;
    scheduleEstimate();
  }

  function exportImage(cb){
    if(!img){ cb(null); return; }
    rebuildFiltered();

    const feather = +featherEl.value||0;
    let maskToUse = maskC;
    if(feather>0){
      const m2 = document.createElement('canvas');
      m2.width = maskC.width; m2.height = maskC.height;
      const m2x = m2.getContext('2d');
      m2x.filter = `blur(${feather}px)`;
      m2x.drawImage(maskC,0,0);
      m2x.filter = 'none';
      maskToUse = m2;
    }

    const finalC = document.createElement('canvas');

    const mx = masked.getContext('2d');
    mx.drawImage(filtC,0,0);
    mx.globalCompositeOperation = 'destination-in';
    mx.drawImage(maskToUse,0,0);
    mx.globalCompositeOperation = 'source-over';

    fx.drawImage(masked,0,0);

    const mime = fmtEl.value;
    const q = (mime==='image/png') ? 1 : (+qEl.value||92)/100;
    finalC.toBlob(b=> cb(b), mime, q);
  }

  // Estimate size throttled
  let estRAF = null;
  function scheduleEstimate(){
    if(estRAF) cancelAnimationFrame(estRAF);
    estRAF = requestAnimationFrame(()=>{
      exportImage(b=>{
        exportBlob = b;
        estInfo.textContent = 'Estimated file size: ' + humanBytes(b?.size||0);
      });
    });
  }

  // Save current mask for undo
  function pushMaskUndo(){
    try{
      maskUndo = maskX.getImageData(0,0,maskC.width,maskC.height);
    }catch(e){ maskUndo = null; }
  }
  function applyUndo(){
    if(maskUndo){
      maskX.putImageData(maskUndo,0,0);
      maskUndo = null;
      composeToPreview();
    }
  }

  // Paint helpers on mask
  function paintDot(x,y,r,erase=false){
    maskX.save();
    maskX.globalCompositeOperation = erase ? 'destination-out' : 'source-over';
    maskX.fillStyle = '#fff';
    maskX.beginPath();
    maskX.arc(x,y,r,0,Math.PI*2);
    maskX.fill();
    maskX.restore();
  }
  function paintLine(x0,y0,x1,y1,r,erase=false){
    maskX.save();
    maskX.globalCompositeOperation = erase ? 'destination-out' : 'source-over';
    maskX.strokeStyle = '#fff';
    maskX.lineWidth = r*2;
    maskX.lineCap = 'round';
    maskX.beginPath();
    maskX.moveTo(x0,y0);
    maskX.lineTo(x1,y1);
    maskX.stroke();
    maskX.restore();
  }
  function commitRect(x,y,w,h){
    maskX.save();
    maskX.fillStyle = '#fff';
    maskX.fillRect(Math.min(x,x+w), Math.min(y,y+h), Math.abs(w), Math.abs(h));
    maskX.restore();
  }
  function commitEllipse(x,y,w,h){
    maskX.save();
    maskX.fillStyle = '#fff';
    maskX.beginPath();
    maskX.ellipse(x+w/2, y+h/2, Math.abs(w/2), Math.abs(h/2), 0, 0, Math.PI*2);
    maskX.fill();
    maskX.restore();
  }

  // Pointer handling
  function startAction(e){
    if(!img) return;
    const pt = ('touches' in e) ? e.touches[0] : e;
    const {ix,iy} = previewToImage(pt.clientX, pt.clientY);
    if(ix<0||iy<0||ix>srcC.width||iy>srcC.height) return;

    pushMaskUndo();

    const tool = toolEl.value;
    const erase = (modePaintEl.value==='erase') || (e.altKey === true);
    if(tool==='brush'){
      painting = true;
      lastPt = {x:ix, y:iy};
      const r = (+brushSizeEl.value||40)/2;
      paintDot(ix,iy,r,erase);
      composeToPreview();
    }else{ // rect / ellipse
      selStart = {x:ix, y:iy};
      selLive = {x:ix, y:iy, w:0, h:0};
      composeToPreview();
    }
    e.preventDefault?.();
  }
  function moveAction(e){
    if(!img) return;
    const pt = ('touches' in e) ? e.touches[0] : e;
    const {ix,iy} = previewToImage(pt.clientX, pt.clientY);

    const tool = toolEl.value;
    if(tool==='brush'){
      if(!painting || !lastPt) return;
      const r = (+brushSizeEl.value||40)/2;
      const erase = (modePaintEl.value==='erase') || (e.altKey === true);
      paintLine(lastPt.x,lastPt.y, ix,iy, r, erase);
      lastPt = {x:ix, y:iy};
      composeToPreview();
    }else{
      if(!selStart) return;
      selLive = {x: selStart.x, y: selStart.y, w: ix - selStart.x, h: iy - selStart.y};
      composeToPreview();
    }
    e.preventDefault?.();
  }
  function endAction(e){
    if(!img) return;
    const tool = toolEl.value;
    if(tool==='brush'){
      painting = false; lastPt = null;
    }else if(selStart && selLive){
      // commit shape to mask
      if(tool==='rect') commitRect(selLive.x, selLive.y, selLive.w, selLive.h);
      else commitEllipse(selLive.x, selLive.y, selLive.w, selLive.h);
      selStart = null; selLive = null;
      composeToPreview();
    }
  }

  // Face detection
  async function detectFaces(){
    if(!img) return alert('Load an image first.');
    if(!('FaceDetector' in window)){
      alert('Face detection not supported in this browser. You can still mask areas manually.');
      return;
    }
    try{
      pushMaskUndo();
      const detector = new FaceDetector({ fastMode:true, maxDetectedFaces: 20 });
      // For speed, run on a downsized bitmap and scale back boxes
      const maxSide = 800;
      const scale = Math.min(1, maxSide / Math.max(srcC.width, srcC.height));
      const smallC = document.createElement('canvas');
      smallC.width = Math.round(srcC.width*scale);
      smallC.height = Math.round(srcC.height*scale);
      smallC.getContext('2d').drawImage(srcC, 0,0, smallC.width, smallC.height);
      const bitmap = await createImageBitmap(smallC);
      const faces = await detector.detect(bitmap);
      const inv = 1/scale;
      const pad = +facePadEl.value||12;

      faces.forEach(f=>{
        const b = f.boundingBox;
        const x = Math.max(0, Math.round(b.x*inv - pad));
        const y = Math.max(0, Math.round(b.y*inv - pad));
        const w = Math.round(b.width*inv + pad*2);
        const h = Math.round(b.height*inv + pad*2);
        commitEllipse(x,y,w,h); // ellipse looks natural on faces
      });
      composeToPreview();
      if(!faces.length) alert('No faces detected. Try manual brush or selections.');
    }catch(err){
      alert('Face detection failed. You can still use brush/selection tools.');
    }
  }

  // Rotation (90° steps) — rotate src and mask together
  function rotateStep(dir){
    if(!img) return;
    const W = srcC.width, H = srcC.height;
    const rC = document.createElement('canvas');
    const rM = document.createElement('canvas');
    if(dir==='L' || dir==='R'){
      rC.width = H; rC.height = W;
      rM.width = H; rM.height = W;
      const rx = rC.getContext('2d'), rm = rM.getContext('2d');
      rx.save();
      rm.save();
      if(dir==='L'){
        rx.translate(0, W); rx.rotate(-Math.PI/2);
        rm.translate(0, W); rm.rotate(-Math.PI/2);
      }else{
        rx.translate(H, 0); rx.rotate(Math.PI/2);
        rm.translate(H, 0); rm.rotate(Math.PI/2);
      }
      rx.drawImage(srcC,0,0);
      rm.drawImage(maskC,0,0);
      rx.restore(); rm.restore();
      srcC = rC; srcX = srcC.getContext('2d');
      maskC = rM; maskX = maskC.getContext('2d');
      filtC.width = srcC.width; filtC.height = srcC.height;
      blendC.width = srcC.width; blendC.height = srcC.height;
      rot = (rot + (dir==='L'?-90:90) + 360) % 360;
      composeToPreview();
    }
  }

  // File handling
  function loadFile(file){
    if(!file || !file.type?.startsWith('image/')) return;
    baseName = (file.name||'image').replace(/\.[^.]+$/,'');
    if(imgURL) URL.revokeObjectURL(imgURL);
    imgURL = URL.createObjectURL(file);
    const im = new Image();
    im.onload = ()=>{
      img = im; rot = 0;
      resetOffscreens(img.width, img.height);
      srcX.imageSmoothingEnabled = true; srcX.imageSmoothingQuality='high';
      srcX.drawImage(img,0,0);
      maskX.clearRect(0,0,maskC.width,maskC.height);
      fileInput.value=''; cameraInput.value='';
      composeToPreview();
    };
    im.onerror = ()=>{ URL.revokeObjectURL(imgURL); fileInput.value=''; cameraInput.value=''; alert('Could not load this image.'); };
    im.src = imgURL;
  }

  // Drag & drop
  ['dragenter','dragover'].forEach(ev=>{
    drop.addEventListener(ev, e=>{ e.preventDefault(); drop.classList.add('drag'); });
  });
  ['dragleave','drop'].forEach(ev=>{
    drop.addEventListener(ev, e=>{ e.preventDefault(); drop.classList.remove('drag'); });
  });
  drop.addEventListener('drop', e=>{
    const f = e.dataTransfer?.files?.[0];
    if(f) loadFile(f);
  });

  // Inputs
  fileInput.addEventListener('change', e=>{
    const f=e.target.files && e.target.files[0]; if(f) loadFile(f);
  });
  cameraInput.addEventListener('change', e=>{
    const f=e.target.files && e.target.files[0]; if(f) loadFile(f);
  });

  // Paste
  window.addEventListener('paste', e=>{
    const items = e.clipboardData?.items;
    if(!items) return;
    for(const it of items){ if(it.type?.startsWith('image/')){ loadFile(it.getAsFile()); break; } }
  });

  // UI wiring
  modeEl.addEventListener('change', ()=>{ setStrengthHint(); if(img) composeToPreview(); });
  [strengthEl, featherEl, brushSizeEl, modePaintEl, showMaskEl].forEach(el=>{
    el.addEventListener('input', ()=>{ if(img) composeToPreview(); });
    el.addEventListener('change', ()=>{ if(img) composeToPreview(); });
  });

  toolEl.addEventListener('change', ()=>{
    brushWrap.classList.toggle('hide', toolEl.value!=='brush');
    overlayMsg.classList.toggle('hide', false);
  });

  rotL.addEventListener('click', ()=> rotateStep('L'));
  rotR.addEventListener('click', ()=> rotateStep('R'));

  undoMaskBtn.addEventListener('click', applyUndo);
  clearMaskBtn.addEventListener('click', ()=>{ if(!img) return; pushMaskUndo(); maskX.clearRect(0,0,maskC.width,maskC.height); composeToPreview(); });

  detectBtn.addEventListener('click', detectFaces);

  // Pointer listeners
  pv.addEventListener('mousedown', startAction);
  window.addEventListener('mousemove', moveAction);
  window.addEventListener('mouseup', endAction);
  pv.addEventListener('touchstart', startAction, {passive:false});
  window.addEventListener('touchmove', moveAction, {passive:false});
  window.addEventListener('touchend', endAction);

  // Download
  downloadBtn.addEventListener('click', ()=>{
    if(!img) return;
    exportImage(blob=>{
      if(!blob) return;
      const extMap = {'image/jpeg':'jpg','image/png':'png','image/webp':'webp'};
      const ext = extMap[fmtEl.value]||'jpg';
      const a = document.createElement('a');
      a.href = URL.createObjectURL(blob);
      a.download = `${baseName}-safe.${ext}`;
      document.body.appendChild(a);
      a.click();
      setTimeout(()=>{ URL.revokeObjectURL(a.href); a.remove(); }, 500);
    });
  });

  // Reset
  resetBtn.addEventListener('click', ()=>{
    if(imgURL){ URL.revokeObjectURL(imgURL); imgURL=null; }
    img=null; baseName='image'; rot=0; exportBlob=null; maskUndo=null;
    resetOffscreens(900,600); // default placeholder size
    srcX.clearRect(0,0,srcC.width,srcC.height);
    maskX.clearRect(0,0,maskC.width,maskC.height);
    strengthEl.value = (modeEl.value==='blur') ? 16 : 20;
    featherEl.value = 6; brushSizeEl.value=50; modePaintEl.value='paint';
    showMaskEl.checked = true; toolEl.value='brush'; brushWrap.classList.remove('hide');
    fmtEl.value='image/jpeg'; qEl.value=92;
    fileInput.value=''; cameraInput.value='';
    ctx.clearRect(0,0,pv.width,pv.height);
    meta.hidden=true; downloadBtn.disabled=true;
    overlayMsg.classList.add('hide');
    composeToPreview();
  });

  // Init
  window.addEventListener('resize', composeToPreview);
  document.getElementById('vt-year').textContent = new Date().getFullYear();
  setStrengthHint();
  overlayMsg.classList.add('hide');
  composeToPreview();
})();
