  function getTargetPixels(){

    const W=parseFloat(wEl.value)||0, H=parseFloat(hEl.value)||0, DPI=parseFloat(dpiEl.value)||300;
    if(units==='px') return {W:Math.round(W), H:Math.round(H)};
    if(units==='cm') return {W:Math.round(W * DPI / 2.54), H:Math.round(H * DPI / 2.54)};
    // inches
    return {W:Math.round(W * DPI), H:Math.round(H * DPI)};
  }
  function setFromPresetPx(W,H){
    unitsEl.value='px'; dpiEl.value=300; wEl.value=W; hEl.value=H; lockEl.checked=false; toggleBlocks(); draw();
  }
  function setFromPresetIn(wIn,hIn,dpi=300){
    unitsEl.value='in'; dpiEl.value=dpi; wEl.value=wIn; hEl.value=hIn; lockEl.checked=false; toggleBlocks(); draw();
  }
  function setFromPresetCm(wCm,hCm,dpi=300){
    unitsEl.value='cm'; dpiEl.value=dpi; wEl.value=wCm; hEl.value=hCm; lockEl.checked=false; toggleBlocks(); draw();
  }
  function humanBytes(bytes){
    if(!bytes) return '—'; const u=['B','KB','MB','GB']; let i=0; let v=bytes;
    while(v>1024&&i<u.length-1){ v/=1024; i++; } return v.toFixed(v<10?2:1)+' '+u[i];
  }

  function currentAspect(){
    if(!img) return null; // use original

  }

  }

      const dy = - (view.cy * s);
      ctx.imageSmoothingEnabled=true; ctx.imageSmoothingQuality='high';
      ctx.drawImage(img, dx, dy, drawW, drawH);
    }

  function roundRect(c,x,y,w,h,r){
    r = Math.min(r, Math.abs(w)/2, Math.abs(h)/2);
    c.beginPath();
    c.moveTo(x+r, y);
    c.arcTo(x+w, y, x+w, y+h, r);
    c.arcTo(x+w, y+h, x, y+h, r);
    c.arcTo(x, y+h, x, y, r);
    c.arcTo(x, y, x+w, y, r);
    c.closePath();
  }

    o.fillStyle = bgEl.value; o.fillRect(0,0,W,H);

    o.translate(W/2, H/2);
    o.rotate(rot*Math.PI/180);

      o.drawImage(img, sx, sy, drawW, drawH);
    }else{
      const baseScale = (modeEl.value==='fill')? Math.max(W/img.width, H/img.height)
                       : Math.min(W/img.width, H/img.height);
      const s = baseScale * ( (+zoomEl.value)/100 );
      const drawW = img.width * s;
      const drawH = img.height * s;
      const dx = - (view.cx * s);
      const dy = - (view.cy * s);
      o.drawImage(img, dx, dy, drawW, drawH);
    }
    o.restore();

    const mime = fmtEl.value;
    const q = (mime==='image/png') ? 1 : (+qEl.value||92)/100;
    off.toBlob(b=> cb(b), mime, q);
  }

  function scheduleEstimate(){
    if(estRAF) cancelAnimationFrame(estRAF);
    estRAF = requestAnimationFrame(()=>{
      exportImage(b=>{ exportBlob=b; estInfo.textContent = 'Estimated file size: '+humanBytes(b?.size||0); });
    });
  }

  // File handling
  function loadFile(file){
    if(!file || !file.type?.startsWith('image/')) return;
    baseName = (file.name||'image').replace(/\.[^.]+$/,'');
    if(imgURL) URL.revokeObjectURL(imgURL);
    imgURL = URL.createObjectURL(file);
    const im = new Image();
    im.onload = ()=>{
      img=im; fileInput.value=''; cameraInput.value='';
      // Set aspect-locked height from width (if checked)
      if(lockEl.checked){ syncAspect('w'); }
      resetView(); draw();
    };
    im.onerror = ()=>{ URL.revokeObjectURL(imgURL); fileInput.value=''; cameraInput.value=''; alert('Could not load this image.'); };
    im.src = imgURL;
  }

  // DnD + Paste
  ['dragenter','dragover'].forEach(ev=> drop.addEventListener(ev, e=>{ e.preventDefault(); drop.classList.add('drag'); }));
  ['dragleave','drop'].forEach(ev=> drop.addEventListener(ev, e=>{ e.preventDefault(); drop.classList.remove('drag'); }));
  drop.addEventListener('drop', e=>{ const f=e.dataTransfer?.files?.[0]; if(f) loadFile(f); });
  window.addEventListener('paste', e=>{
    const items = e.clipboardData?.items||[];
    for(const it of items){ if(it.type?.startsWith('image/')){ loadFile(it.getAsFile()); break; } }
  });

  // Inputs
  fileInput.addEventListener('change', e=>{ const f=e.target.files?.[0]; if(f) loadFile(f); });
  cameraInput.addEventListener('change', e=>{ const f=e.target.files?.[0]; if(f) loadFile(f); });

  unitsEl.addEventListener('change', ()=>{ toggleBlocks(); draw(); });
  [wEl,hEl,dpiEl].forEach(el=>{
    el.addEventListener('input', (e)=>{ if(e.target===wEl) syncAspect('w'); if(e.target===hEl) syncAspect('h'); draw(); });
    el.addEventListener('change', draw);
  });
  lockEl.addEventListener('change', ()=>{ if(lockEl.checked) syncAspect('w'); draw(); });

  [modeEl,bgEl].forEach(el=>{ el.addEventListener('input', draw); el.addEventListener('change', draw); });
  [fmtEl,qEl].forEach(el=>{
    el.addEventListener('input', ()=>{ if(img) scheduleEstimate(); });
    el.addEventListener('change', ()=>{ if(img) scheduleEstimate(); });
  });

  zoomEl.addEventListener('input', draw);

  rotL.addEventListener('click', ()=>{ rot=(rot-90+360)%360; resetView(); draw(); });
  rotR.addEventListener('click', ()=>{ rot=(rot+90)%360; resetView(); draw(); });
  flipH.addEventListener('click', ()=>{ flip*=-1; draw(); });

  // Presets
  document.querySelectorAll('[data-preset]').forEach(b=>{
    b.addEventListener('click', ()=>{
      const p=b.dataset.preset;
      if(p==='1080x1080') setFromPresetPx(1080,1080);
      if(p==='1920x1080') setFromPresetPx(1920,1080);
      if(p==='1x1in') setFromPresetIn(1,1,300);
      if(p==='2x2in') setFromPresetIn(2,2,300);
      if(p==='3.5x4.5cm') setFromPresetCm(3.5,4.5,300);
      if(p==='4x6in') setFromPresetIn(4,6,300);
    });
  });

  // Drag to pan
  function startDrag(e){
    if(!img) return;
    dragging=true;
    const pt=('touches' in e)? e.touches[0]:e;
    lastX=pt.clientX; lastY=pt.clientY;
    e.preventDefault?.();
  }
  function moveDrag(e){
    if(!dragging||!img) return;
    const pt=('touches' in e)? e.touches[0]:e;
    const dx=pt.clientX-lastX, dy=pt.clientY-lastY;
    lastX=pt.clientX; lastY=pt.clientY;

    const {W,H}=getTargetPixels();
    const frameW = pv.width * 0.88;
    const scalePreview = frameW / W;
    // Rotate-aware pan
    const ang = -rot*Math.PI/180;
    const rx =  dx*Math.cos(ang) - dy*Math.sin(ang);
    const ry =  dx*Math.sin(ang) + dy*Math.cos(ang);
    const userScale = (+zoomEl.value)/100;
    const baseScale = (modeEl.value==='fill')? Math.max(W/img.width, H/img.height)
                     : (modeEl.value==='fit') ? Math.min(W/img.width, H/img.height)
                     : (W/img.width);

    let s;
    if(modeEl.value==='stretch'){
      s = 1 * scalePreview; // since we draw W×H directly
    }else{
      s = baseScale * userScale * scalePreview;
    }

    view.cx -= rx / (s * (1/flip));
    view.cy -= ry / s;

    // clamp loosely
    view.cx = Math.max(-img.width, Math.min(2*img.width, view.cx));
    view.cy = Math.max(-img.height, Math.min(2*img.height, view.cy));
    draw();
  }
  function endDrag(){ dragging=false; }

  pv.addEventListener('mousedown', startDrag);
  window.addEventListener('mousemove', moveDrag);
  window.addEventListener('mouseup', endDrag);
  pv.addEventListener('touchstart', startDrag, {passive:false});
  window.addEventListener('touchmove', moveDrag, {passive:false});
  window.addEventListener('touchend', endDrag);

  // Download
  downloadBtn.addEventListener('click', ()=>{
    if(!img) return;
    exportImage(blob=>{
      if(!blob) return;
      const extMap={'image/jpeg':'jpg','image/png':'png','image/webp':'webp'};
      const ext=extMap[fmtEl.value]||'jpg';
      const a=document.createElement('a');
      a.href=URL.createObjectURL(blob);
      a.download=`${baseName}-resized.${ext}`;
      document.body.appendChild(a); a.click();
      setTimeout(()=>{ URL.revokeObjectURL(a.href); a.remove(); }, 600);
    });
  });

  // Reset
  resetBtn.addEventListener('click', ()=>{
    if(imgURL){ URL.revokeObjectURL(imgURL); imgURL=null; }
    img=null; baseName='image'; rot=0; flip=1;
    unitsEl.value='px'; wEl.value=600; hEl.value=400; dpiEl.value=300; lockEl.checked=true;
    modeEl.value='fit'; bgEl.value='#ffffff';
    zoomEl.value=100; fmtEl.value='image/jpeg'; qEl.value=92;
    fileInput.value=''; cameraInput.value='';
    ctx.clearRect(0,0,pv.width,pv.height);
    meta.hidden=true; downloadBtn.disabled=true;
    toggleBlocks();
  });

  // UI toggles
  function toggleBlocks(){
    dpiWrap.classList.toggle('hide', unitsEl.value==='px');
  }

  // Init
  window.addEventListener('resize', draw);
  document.getElementById('vt-year').textContent = new Date().getFullYear();
  toggleBlocks(); draw();
})();
