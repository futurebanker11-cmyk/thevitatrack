    customCX = img.width/2; customCY = img.height/2;
  }

  function getPreviewScale(){
    // Scale to fit image into pv while preserving aspect
    const s = Math.min(pv.width / (img?.width||pv.width), pv.height / (img?.height||pv.height));
    return s || 1;
  }

  function draw(){
    fitCanvas();
    ctx.clearRect(0,0,pv.width,pv.height);
    if(!img){ meta.hidden=true; downloadBtn.disabled=true; return; }

    const scale = getPreviewScale();
    const drawW = img.width * scale;
    const drawH = img.height * scale;
    const dx = (pv.width - drawW)/2;
    const dy = (pv.height - drawH)/2;

    // Base image
    ctx.imageSmoothingEnabled = true;
    ctx.imageSmoothingQuality = 'high';
    ctx.drawImage(img, dx, dy, drawW, drawH);

    // Watermark
    const type = wmTypeEl.value;
    const pmode = placeModeEl.value;

    if(type==='text'){
      drawTextWM(ctx, scale, dx, dy, pmode);
    }else{
      drawLogoWM(ctx, scale, dx, dy, pmode);
    }

    // Meta
    origInfo.textContent = `Original: ${img.width}×${img.height}px`;
    expInfo.textContent = `Export: ${img.width}×${img.height}px`;
    meta.hidden = false;

    scheduleEstimate();
    downloadBtn.disabled = false;
  }

  // ==== TEXT WM ====
  function textStyle(ctx2, pxScale=1){
    const fsize = +fontSizeEl.value||48;
    ctx2.font = `${fsize*pxScale}px ${fontFamilyEl.value}`;
    ctx2.fillStyle = textColorEl.value;
    const sh = shadowEl.value;
    if(sh==='soft'){ ctx2.shadowColor = 'rgba(0,0,0,.35)'; ctx2.shadowBlur = 2*pxScale; }
    else if(sh==='strong'){ ctx2.shadowColor = 'rgba(0,0,0,.55)'; ctx2.shadowBlur = 4*pxScale; }
    else { ctx2.shadowColor = 'transparent'; ctx2.shadowBlur = 0; }
  }
  function measureText(ctx2, text){
    const m = ctx2.measureText(text);
    const ascent = m.actualBoundingBoxAscent || 0;
    const descent = m.actualBoundingBoxDescent || 0;
    return { w: m.width, h: ascent + descent };
  }
  function getSingleAnchor(imgW,imgH, unitW,unitH, padding, pos){
    const pad = padding;
    switch(pos){
      case 'top-left': return { cx: pad + unitW/2, cy: pad + unitH/2 };
      case 'top-right': return { cx: imgW - pad - unitW/2, cy: pad + unitH/2 };
      case 'bottom-left': return { cx: pad + unitW/2, cy: imgH - pad - unitH/2 };
      case 'bottom-right': return { cx: imgW - pad - unitW/2, cy: imgH - pad - unitH/2 };
      case 'center': return { cx: imgW/2, cy: imgH/2 };
      case 'custom': default: return { cx: customCX||imgW/2, cy: customCY||imgH/2 };
    }
  }
  function drawTextWM(ctx2, scale, dx, dy, pmode){
    const text = wmTextEl.value || '';
    if(!text) return;
    const opa = (+opacityEl.value||30)/100;
    const rot = (+rotationEl.value||0) * Math.PI/180;
    const pad = +paddingEl.value||0;

    // Measure in base image pixels
    const tmp = document.createElement('canvas').getContext('2d');
    textStyle(tmp, 1);
    const {w:tw, h:th} = measureText(tmp, text);

    if(pmode==='single'){
      const pos = positionEl.value;
      const anchor = getSingleAnchor(img.width, img.height, tw, th, pad, pos);
      // Draw at preview scale
      ctx2.save();
      ctx2.globalAlpha = opa;
      textStyle(ctx2, scale);
      ctx2.translate(dx + anchor.cx*scale, dy + anchor.cy*scale);
      ctx2.rotate(rot);
      ctx2.textAlign = 'center';
      ctx2.textBaseline = 'middle';
      ctx2.fillText(text, 0, 0);
      ctx2.restore();
    }else{ // tile
      const angle = (+tileAngleEl.value||0) * Math.PI/180;
      const gapX = +tileGapXEl.value||140;
      const gapY = +tileGapYEl.value||120;
      const stagger = tileOffsetEl.value;

      // Unit size in base pixels
      const unitW = tw, unitH = th;
      // Step distances (base pixels)
      const stepX = unitW + gapX;
      const stepY = unitH + gapY;

      // Start beyond canvas to cover rotated edges
      const startX = -unitW*2, endX = img.width + unitW*2;
      const startY = -unitH*2, endY = img.height + unitH*2;

      ctx2.save();
      ctx2.globalAlpha = opa;
      textStyle(ctx2, scale);
      ctx2.textAlign = 'center';
      ctx2.textBaseline = 'middle';

      for(let y=startY, row=0; y<=endY; y+=stepY, row++){
        const offset = (stagger==='half') ? (stepX/2 * (row%2)) : 0;
        for(let x=startX+offset; x<=endX; x+=stepX){
          const px = dx + x*scale;
          const py = dy + y*scale;
          ctx2.save();
          ctx2.translate(px, py);
          ctx2.rotate(angle);
          ctx2.fillText(text, 0, 0);
          ctx2.restore();
        }
      }
      ctx2.restore();
    }
  }

  // ==== LOGO WM ====
  function drawLogoWM(ctx2, scale, dx, dy, pmode){
    if(!logo) return;
    const opa = (+logoOpacityEl.value||30)/100;
    const rot = (+logoRotationEl.value||0) * Math.PI/180;
    const pad = +logoPaddingEl.value||0;

    // Logo base size in image pixels: width = % of image width
    const targetW = (Math.max(5, +logoScaleEl.value||20) / 100) * img.width;
    const ratio = logo.height ? (logo.height/logo.width) : 1;
    const targetH = targetW * ratio;

    if(pmode==='single'){
      const pos = positionEl.value;
      const anchor = getSingleAnchor(img.width, img.height, targetW, targetH, pad, pos);
      ctx2.save();
      ctx2.globalAlpha = opa;
      ctx2.translate(dx + anchor.cx*scale, dy + anchor.cy*scale);
      ctx2.rotate(rot);
      ctx2.drawImage(logo, -targetW*scale/2, -targetH*scale/2, targetW*scale, targetH*scale);
      ctx2.restore();
    }else{ // tile
      const angle = (+tileAngleImgEl.value||0) * Math.PI/180;
      const gapX = +tileGapXImgEl.value||160;
      const gapY = +tileGapYImgEl.value||140;
      const stagger = tileOffsetImgEl.value;

      const unitW = targetW, unitH = targetH;
      const stepX = unitW + gapX;
      const stepY = unitH + gapY;

      const startX = -unitW*2, endX = img.width + unitW*2;
      const startY = -unitH*2, endY = img.height + unitH*2;

      ctx2.save();
      ctx2.globalAlpha = opa;

      for(let y=startY, row=0; y<=endY; y+=stepY, row++){
        const offset = (stagger==='half') ? (stepX/2 * (row%2)) : 0;
        for(let x=startX+offset; x<=endX; x+=stepX){
          const px = dx + x*scale;
          const py = dy + y*scale;
          ctx2.save();
          ctx2.translate(px, py);
          ctx2.rotate(angle);
          ctx2.drawImage(logo, -unitW*scale/2, -unitH*scale/2, unitW*scale, unitH*scale);
          ctx2.restore();
        }
      }
      ctx2.restore();
    }
  }

  // ===== Export =====
  function exportImage(cb){
    if(!img){ cb(null); return; }
    const off = document.createElement('canvas');
    off.width = img.width; off.height = img.height;
    const octx = off.getContext('2d');
    octx.imageSmoothingEnabled = true; octx.imageSmoothingQuality = 'high';
    octx.drawImage(img, 0, 0);

    const type = wmTypeEl.value;
    const pmode = placeModeEl.value;

    if(type==='text'){
      // Prepare text style for 1:1 base pixels
      const tmp = document.createElement('canvas').getContext('2d');
      tmp.font = `${+fontSizeEl.value||48}px ${fontFamilyEl.value}`;
      const text = wmTextEl.value||'';
      const m = tmp.measureText(text);
      const unitW = m.width, unitH = (m.actualBoundingBoxAscent||0)+(m.actualBoundingBoxDescent||0);
      const opa = (+opacityEl.value||30)/100;
      const rot = (+rotationEl.value||0) * Math.PI/180;
      const pad = +paddingEl.value||0;

      if(pmode==='single'){
        const pos = positionEl.value;
        const anchor = getSingleAnchor(off.width, off.height, unitW, unitH, pad, pos);
        octx.save();
        octx.globalAlpha = opa;
        octx.font = `${+fontSizeEl.value||48}px ${fontFamilyEl.value}`;
        // shadow
        const sh = shadowEl.value;
        if(sh==='soft'){ octx.shadowColor='rgba(0,0,0,.35)'; octx.shadowBlur=2; }
        else if(sh==='strong'){ octx.shadowColor='rgba(0,0,0,.55)'; octx.shadowBlur=4; }
        else { octx.shadowColor='transparent'; octx.shadowBlur=0; }
        octx.fillStyle = textColorEl.value;
        octx.textAlign='center'; octx.textBaseline='middle';
        octx.translate(anchor.cx, anchor.cy);
        octx.rotate(rot);
        octx.fillText(text, 0, 0);
        octx.restore();
      }else{
        const angle = (+tileAngleEl.value||0) * Math.PI/180;
        const gapX = +tileGapXEl.value||140;
        const gapY = +tileGapYEl.value||120;
        const stagger = tileOffsetEl.value;

        const stepX = unitW + gapX;
        const stepY = unitH + gapY;
        const startX = -unitW*2, endX = off.width + unitW*2;
        const startY = -unitH*2, endY = off.height + unitH*2;

        octx.save();
        octx.globalAlpha = opa;
        octx.font = `${+fontSizeEl.value||48}px ${fontFamilyEl.value}`;
        const sh = shadowEl.value;
        if(sh==='soft'){ octx.shadowColor='rgba(0,0,0,.35)'; octx.shadowBlur=2; }
        else if(sh==='strong'){ octx.shadowColor='rgba(0,0,0,.55)'; octx.shadowBlur=4; }
        else { octx.shadowColor='transparent'; octx.shadowBlur=0; }
        octx.fillStyle = textColorEl.value;
        octx.textAlign='center'; octx.textBaseline='middle';

        for(let y=startY, row=0; y<=endY; y+=stepY, row++){
          const offset = (stagger==='half') ? (stepX/2 * (row%2)) : 0;
          for(let x=startX+offset; x<=endX; x+=stepX){
            octx.save();
            octx.translate(x, y);
            octx.rotate(angle);
            octx.fillText(text, 0, 0);
            octx.restore();
          }
        }
        octx.restore();
      }
    }else{
      if(!logo){ cb(null); return; }
      const targetW = (Math.max(5, +logoScaleEl.value||20) / 100) * off.width;
      const ratio = logo.height ? (logo.height/logo.width) : 1;
      const targetH = targetW * ratio;
      const opa = (+logoOpacityEl.value||30)/100;
      const pad = +logoPaddingEl.value||0;

      if(placeModeEl.value==='single'){
        const pos = positionEl.value;
        const anchor = getSingleAnchor(off.width, off.height, targetW, targetH, pad, pos);
        octx.save();
        octx.globalAlpha = opa;
        octx.translate(anchor.cx, anchor.cy);
        octx.rotate((+logoRotationEl.value||0)*Math.PI/180);
        octx.drawImage(logo, -targetW/2, -targetH/2, targetW, targetH);
        octx.restore();
      }else{
        const angle = (+tileAngleImgEl.value||0) * Math.PI/180;
        const gapX = +tileGapXImgEl.value||160;
        const gapY = +tileGapYImgEl.value||140;
        const stagger = tileOffsetImgEl.value;

        const stepX = targetW + gapX;
        const stepY = targetH + gapY;
        const startX = -targetW*2, endX = off.width + targetW*2;
        const startY = -targetH*2, endY = off.height + targetH*2;

        octx.save();
        octx.globalAlpha = opa;
        for(let y=startY, row=0; y<=endY; y+=stepY, row++){

            octx.save();
            octx.translate(x, y);
            octx.rotate(angle);
            octx.drawImage(logo, -targetW/2, -targetH/2, targetW, targetH);
            octx.restore();
          }
        }
        octx.restore();
      }
    }

    const mime = fmtEl.value;
    const q = (mime==='image/png') ? 1 : (+qEl.value||92)/100;
    off.toBlob(b=> cb(b), mime, q);
  }

  // Drag to set custom position (single/custom)
  function previewToImageCoords(clientX, clientY){
    const rect = pv.getBoundingClientRect();
    const x = clientX - rect.left;
    const y = clientY - rect.top;
    const scale = getPreviewScale();
    const dx = (pv.width - img.width*scale)/2;
    const dy = (pv.height - img.height*scale)/2;
    const ix = (x - dx) / scale;
    const iy = (y - dy) / scale;
    return {ix, iy};
  }
  function startDrag(e){
    if(placeModeEl.value!=='single' || positionEl.value!=='custom' || !img) return;
    dragging = true;
    const pt = ('touches' in e) ? e.touches[0] : e;
    const {ix,iy} = previewToImageCoords(pt.clientX, pt.clientY);
    lastX = ix; lastY = iy;
    e.preventDefault?.();
  }
  function moveDrag(e){
    if(!dragging || !img) return;
    const pt = ('touches' in e) ? e.touches[0] : e;
    const {ix,iy} = previewToImageCoords(pt.clientX, pt.clientY);
    const dx = ix - lastX, dy = iy - lastY;
    customCX = Math.max(0, Math.min(img.width, customCX + dx));
    customCY = Math.max(0, Math.min(img.height, customCY + dy));
    lastX = ix; lastY = iy;
    draw();
  }
  function endDrag(){ dragging = false; }

  pv.addEventListener('mousedown', startDrag);
  window.addEventListener('mousemove', moveDrag);
  window.addEventListener('mouseup', endDrag);
  pv.addEventListener('touchstart', startDrag, {passive:false});
  window.addEventListener('touchmove', moveDrag, {passive:false});
  window.addEventListener('touchend', endDrag);

    tileAngleEl.value=-30; tileGapXEl.value=140; tileGapYEl.value=120; tileOffsetEl.value='half';
    logoScaleEl.value=22; logoOpacityEl.value=28; logoRotationEl.value=-25; logoPaddingEl.value=16;
    tileAngleImgEl.value=-30; tileGapXImgEl.value=160; tileGapYImgEl.value=140; tileOffsetImgEl.value='half';
    fmtEl.value='image/jpeg'; qEl.value=92;
