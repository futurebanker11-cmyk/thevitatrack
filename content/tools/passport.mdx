<script>
(() => {
  const $ = s => document.querySelector(s);
  const out = $('#out'), ctx = out.getContext('2d');
  const drop = $('#drop'), fileInput = $('#file'), cameraInput = $('#cameraFile');
  const zoomEl = $('#zoom'), featherEl = $('#feather'), dpiEl = $('#dpi');
  const countryEl = $('#country'), bgEl = $('#bg'), fmtEl = $('#format');
  const downloadBtn = $('#downloadBtn'), resetBtn = $('#resetBtn');
  const presetsWrap = $('#presets');
  const centerBtn = $('#centerBtn');
  const meta = $('#meta'), sizeInfo = $('#sizeInfo'), dpiInfo = $('#dpiInfo'), estInfo = $('#estInfo');

  const inchPerMm = 0.03937007874;

  let img = null, imgURL = null, baseName = 'photo';
  let frameWpx = 900, frameHpx = 1200; // default 3:4
  let zoom = 100;
  let offsetX = 0, offsetY = 0;
  let dragging = false, lastX = 0, lastY = 0;
  let renderedBlob = null;

  // Country size presets (width_mm, height_mm, and rough eye/chin guides)
  const PRESETS = {
    US: { mm:[50.8, 50.8], note:'2×2 in', eyeY: 0.42, chinY: 0.70 },
    EU: { mm:[35, 45], note:'35×45 mm', eyeY: 0.40, chinY: 0.70 },
    UK: { mm:[35, 45], note:'35×45 mm', eyeY: 0.40, chinY: 0.70 },
    IN: { mm:[35, 45], note:'35×45 mm', eyeY: 0.40, chinY: 0.70 },
  };

  function mmToPx(mm, dpi){ return Math.round(mm * inchPerMm * dpi); }

  function setFrameFromPreset(code){
    const p = PRESETS[code] || PRESETS.EU;
    const dpi = +dpiEl.value || 300;
    frameWpx = mmToPx(p.mm[0], dpi);
    frameHpx = mmToPx(p.mm[1], dpi);
    out.width = frameWpx;
    out.height = frameHpx;
    // Reset positioning/zoom
    zoom = 100; offsetX = 0; offsetY = 0; zoomEl.value = 100;
    draw();
    updateMeta();
  }

  function updateMeta(){
    const code = countryEl.value;
    const p = PRESETS[code];
    sizeInfo.textContent = `Size: ${p.mm[0]}×${p.mm[1]} mm (${frameWpx}×${frameHpx}px)`;
    dpiInfo.textContent = `DPI: ${dpiEl.value}`;
    if(renderedBlob) estInfo.textContent = 'Estimated file size: ' + humanBytes(renderedBlob.size);
    meta.hidden = false;
  }

  function humanBytes(bytes){
    if(!bytes) return '—';
    const u=['B','KB','MB','GB']; let i=0; let v=bytes;
    while(v>1024 && i<u.length-1){ v/=1024; i++; }
    return v.toFixed(v<10?2:1)+' '+u[i];
  }

  function clearCanvas(color){
    ctx.save();
    ctx.fillStyle = color || '#fff';
    ctx.fillRect(0,0,out.width,out.height);
    ctx.restore();
  }

  function draw(){
    clearCanvas(bgEl.value);
    if(!img){
      ctx.save(); ctx.fillStyle = '#e5e7eb'; ctx.fillRect(0,0,out.width,out.height); ctx.restore();
      placeGuides(); return;
    }
    const imgW = img.naturalWidth || img.width;
    const imgH = img.naturalHeight || img.height;

    const scaleBase = Math.max(out.width / imgW, out.height / imgH);
    const scale = scaleBase * (zoom/100);

    const drawW = imgW * scale;
    const drawH = imgH * scale;
    const cx = out.width/2 + offsetX;
    const cy = out.height/2 + offsetY;

    const dx = Math.round(cx - drawW/2);
    const dy = Math.round(cy - drawH/2);

    ctx.imageSmoothingEnabled = true;
    ctx.imageSmoothingQuality = 'high';

    // Feather blend (soften edges to background)
    const feather = +featherEl.value || 0;
    if(feather>0){
      ctx.drawImage(img, dx, dy, drawW, drawH);
      ctx.save();
      ctx.globalAlpha = Math.min(0.6, feather/100);
      ctx.fillStyle = bgEl.value;
      ctx.fillRect(0,0,out.width,out.height);
      ctx.restore();
      ctx.drawImage(img, dx, dy, drawW, drawH);
    }else{
      ctx.drawImage(img, dx, dy, drawW, drawH);
    }

    // Save blob estimate
    const mime = fmtEl.value;
    const quality = (mime==='image/png') ? 1 : 0.92;
    out.toBlob(b=>{ renderedBlob=b; updateMeta(); }, mime, quality);

    placeGuides();
  }

  function placeGuides(){
    const code = countryEl.value;
    const p = PRESETS[code];
    const eyeLine = $('#eyeLine'), chinLine = $('#chinLine');
    eyeLine.style.top = (p.eyeY*100) + '%';
    chinLine.style.top = (p.chinY*100) + '%';
  }

  // File handling
  function loadFile(file){
    if(!file || !file.type?.startsWith('image/')) return;
    baseName = (file.name||'photo').replace(/\.[^.]+$/,'');
    if(imgURL) URL.revokeObjectURL(imgURL);
    imgURL = URL.createObjectURL(file);
    const im = new Image();
    im.onload = ()=>{ img = im; fileInput.value=''; cameraInput.value=''; draw(); enableDownload(); };
    im.onerror = ()=>{ URL.revokeObjectURL(imgURL); fileInput.value=''; cameraInput.value=''; alert('Could not load this image.'); };
    im.src = imgURL;
  }

  // Drag/drop
  ['dragenter','dragover'].forEach(ev=>{
    drop.addEventListener(ev, e=>{ e.preventDefault(); drop.classList.add('drag'); });
  });
  ['dragleave','drop'].forEach(ev=>{
    drop.addEventListener(ev, e=>{ e.preventDefault(); drop.classList.remove('drag'); });
  });
  drop.addEventListener('drop', e=>{
    const f = e.dataTransfer?.files?.[0];
    if(f) loadFile(f);
  });

  // Real inputs (mobile-safe)
  fileInput.addEventListener('change', e=>{
    const f = e.target.files && e.target.files[0];
    if(f) loadFile(f);
  });
  cameraInput.addEventListener('change', e=>{
    const f = e.target.files && e.target.files[0];
    if(f) loadFile(f);
  });

  // Paste
  window.addEventListener('paste', e=>{
    const items = e.clipboardData?.items;
    if(!items) return;
    for(const it of items){
      if(it.type?.startsWith('image/')){ loadFile(it.getAsFile()); break; }
    }
  });

  // Zoom / background / format
  zoomEl.addEventListener('input', ()=>{ zoom = +zoomEl.value || 100; draw(); });
  [bgEl, fmtEl].forEach(el=> el.addEventListener('change', draw));
  featherEl.addEventListener('input', draw);
  featherEl.addEventListener('change', draw);

  // DPI / Country changes
  function applyPreset(){ setFrameFromPreset(countryEl.value); }
  countryEl.addEventListener('change', applyPreset);
  dpiEl.addEventListener('change', applyPreset);

  // Preset chips
  presetsWrap.addEventListener('click', e=>{
    const b = e.target.closest('.chip'); if(!b) return;
    countryEl.value = b.dataset.preset;
    applyPreset();
  });

  // Drag to move
  const wrap = $('#canvasWrap');
  wrap.addEventListener('mousedown', e=>{
    dragging=true; lastX=e.clientX; lastY=e.clientY;
  });
  window.addEventListener('mousemove', e=>{
    if(!dragging) return;
    offsetX += (e.clientX - lastX);
    offsetY += (e.clientY - lastY);
    lastX = e.clientX; lastY = e.clientY;
    draw();
  });
  window.addEventListener('mouseup', ()=> dragging=false);

  // Touch drag
  wrap.addEventListener('touchstart', e=>{
    if(e.touches.length!==1) return;
    const t=e.touches[0]; dragging=true; lastX=t.clientX; lastY=t.clientY;
  }, {passive:true});
  window.addEventListener('touchmove', e=>{
    if(!dragging || e.touches.length!==1) return;
    const t=e.touches[0];
    offsetX += (t.clientX - lastX);
    offsetY += (t.clientY - lastY);
    lastX = t.clientX; lastY = t.clientY;
    draw();
  }, {passive:true});
  window.addEventListener('touchend', ()=> dragging=false);

  // Nudge buttons
  document.querySelectorAll('[data-nudge]').forEach(btn=>{
    btn.addEventListener('click', ()=>{
      const step = Math.max(1, Math.round(Math.min(out.width,out.height)/100));
      const dir = btn.dataset.nudge;
      if(dir==='up') offsetY -= step;
      if(dir==='down') offsetY += step;
      if(dir==='left') offsetX -= step;
      if(dir==='right') offsetX += step;
      draw();
    });
  });
  centerBtn.addEventListener('click', ()=>{ offsetX=0; offsetY=0; draw(); });

  // Download
  function enableDownload(){ downloadBtn.disabled = !img; }
  downloadBtn.addEventListener('click', ()=>{
    if(!img) return;
    const extMap = {'image/jpeg':'jpg','image/png':'png','image/webp':'webp'};
    const ext = extMap[fmtEl.value] || 'jpg';
    out.toBlob(b=>{
      if(!b) return;
      const a = document.createElement('a');
      a.href = URL.createObjectURL(b);
      a.download = `${baseName}-passport.${ext}`;
      document.body.appendChild(a);
      a.click();
      setTimeout(()=>{ URL.revokeObjectURL(a.href); a.remove(); }, 500);
    }, fmtEl.value, fmtEl.value==='image/png' ? 1 : 0.92);
  });

  // Reset
  resetBtn.addEventListener('click', ()=>{
    img = null; if(imgURL) { URL.revokeObjectURL(imgURL); imgURL=null; }
    baseName='photo'; zoom=100; offsetX=offsetY=0; zoomEl.value=100; featherEl.value=20;
    countryEl.value='US'; dpiEl.value=300; bgEl.value='#ffffff'; fmtEl.value='image/jpeg';
    setFrameFromPreset('US');
    renderedBlob=null; meta.hidden=true;
    draw(); enableDownload();
  });

  // Init
  setFrameFromPreset('US');
  document.getElementById('vt-year').textContent = new Date().getFullYear();
})();
