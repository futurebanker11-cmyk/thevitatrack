    if(estRAF) cancelAnimationFrame(estRAF);
    estRAF = requestAnimationFrame(()=>{
      exportBanner(OUT_W, OUT_H, b=>{ exportBlob=b; estInfo.textContent = 'Estimated file size: '+humanBytes(b?.size||0); });
    });
  }

  // ===== Interaction: drag to pan =====
  function startDrag(e){
    if(!img) return;
    dragging=true;
    const pt=('touches' in e)? e.touches[0]:e;
    lastX=pt.clientX; lastY=pt.clientY;
    e.preventDefault?.();
  }
  function moveDrag(e){
    if(!dragging||!img) return;
    const pt=('touches' in e)? e.touches[0]:e;
    const dx=pt.clientX-lastX, dy=pt.clientY-lastY;
    lastX=pt.clientX; lastY=pt.clientY;

    // Map screen delta to image space
    const frameW = pv.width * 0.92;
    const CH = Math.min(frameW * (OUT_H/OUT_W), pv.height*0.80);
    const CW = CH * (OUT_W/OUT_H);
    const scalePreview = CW / OUT_W;

    const sFit = Math.min(OUT_W/img.width, OUT_H/img.height);
    const sFill = Math.max(OUT_W/img.width, OUT_H/img.height);
    const baseScale = (modeEl.value==='fill')? sFill : sFit;
    const userScale = (+zoomEl.value)/100;
    const s = baseScale * userScale * scalePreview;

    const ang = -rot*Math.PI/180;
    const rx =  dx*Math.cos(ang) - dy*Math.sin(ang);
    const ry =  dx*Math.sin(ang) + dy*Math.cos(ang);

    view.cx -= rx / (s * (1/flip));
    view.cy -= ry / s;

    // Loose clamp
    view.cx = Math.max(-img.width, Math.min(2*img.width, view.cx));
    view.cy = Math.max(-img.height, Math.min(2*img.height, view.cy));
    draw();
  }
  function endDrag(){ dragging=false; }

  pv.addEventListener('mousedown', startDrag);
  window.addEventListener('mousemove', moveDrag);
  window.addEventListener('mouseup', endDrag);
  pv.addEventListener('touchstart', startDrag, {passive:false});
  window.addEventListener('touchmove', moveDrag, {passive:false});
  window.addEventListener('touchend', endDrag);

  // ===== File handling =====
  function loadFile(file){
    if(!file || !file.type?.startsWith('image/')) return;
    baseName = (file.name||'image').replace(/\.[^.]+$/,'');
    if(imgURL) URL.revokeObjectURL(imgURL);
    imgURL = URL.createObjectURL(file);
    const im = new Image();
    im.onload = ()=>{
      img=im; fileInput.value=''; cameraInput.value='';
      resetView(); draw();
    };
    im.onerror = ()=>{ URL.revokeObjectURL(imgURL); fileInput.value=''; cameraInput.value=''; alert('Could not load this image.'); };
    im.src = imgURL;
  }

  // Change handlers (fires after user picks)
  fileInput.addEventListener('change', e=>{ const f=e.target.files?.[0]; if(f) loadFile(f); });
  cameraInput.addEventListener('change', e=>{ const f=e.target.files?.[0]; if(f) loadFile(f); });

  // Drag & drop
  ['dragenter','dragover'].forEach(ev=> drop.addEventListener(ev, e=>{ e.preventDefault(); drop.classList.add('drag'); }));
  ['dragleave','drop'].forEach(ev=> drop.addEventListener(ev, e=>{ e.preventDefault(); drop.classList.remove('drag'); }));
  drop.addEventListener('drop', e=>{ const f=e.dataTransfer?.files?.[0]; if(f) loadFile(f); });

  // Paste
  window.addEventListener('paste', e=>{
    const items = e.clipboardData?.items||[];
    for(const it of items){ if(it.type?.startsWith('image/')){ loadFile(it.getAsFile()); break; } }
  });

  // Controls
  [modeEl, bgEl, guidesEl, overlayOpacityEl].forEach(el=>{ el.addEventListener('input', draw); el.addEventListener('change', draw); });
  zoomEl.addEventListener('input', draw);
  rotL.addEventListener('click', ()=>{ rot=(rot-90+360)%360; draw(); });
  rotR.addEventListener('click', ()=>{ rot=(rot+90)%360; draw(); });
  flipH.addEventListener('click', ()=>{ flip*=-1; draw(); });

  // Presets
  document.querySelectorAll('[data-preset]').forEach(b=>{
    b.addEventListener('click', ()=>{
      const p=b.dataset.preset;
      if(p==='tight'){ zoomEl.value = Math.min(200, (+zoomEl.value||100)+40); }
      if(p==='wide'){ zoomEl.value = Math.max(50, (+zoomEl.value||100)-30); }
      if(p==='center' && img){ view.cx = img.width/2; view.cy = img.height/2; }
      draw();
    });
  });

  // Downloads
  function triggerDownload(blob, name){
    const a=document.createElement('a');
    a.href=URL.createObjectURL(blob); a.download=name;
    document.body.appendChild(a); a.click();
    setTimeout(()=>{ URL.revokeObjectURL(a.href); a.remove(); }, 500);
  }
  downloadFull.addEventListener('click', ()=>{
    if(!img) return;
    exportBanner(OUT_W, OUT_H, b=>{ if(!b) return;
      const extMap={'image/jpeg':'jpg','image/png':'png','image/webp':'webp'};
      triggerDownload(b, `${baseName}-youtube-banner-2560x1440.${extMap[fmtEl.value]||'jpg'}`);
    });
  });
  downloadDesktop.addEventListener('click', ()=>{
    if(!img) return;
    exportBanner(OUT_W, DESKTOP_H, b=>{ if(!b) return;
      const extMap={'image/jpeg':'jpg','image/png':'png','image/webp':'webp'};
      triggerDownload(b, `${baseName}-desktop-visible-2560x423.${extMap[fmtEl.value]||'jpg'}`);
    });
  });
  downloadSafe.addEventListener('click', ()=>{
    if(!img) return;
    exportBanner(SAFE_W, SAFE_H, b=>{ if(!b) return;
      const extMap={'image/jpeg':'jpg','image/png':'png','image/webp':'webp'};
      triggerDownload(b, `${baseName}-all-devices-safe-1546x423.${extMap[fmtEl.value]||'jpg'}`);
    });
  });

  // Estimate
  [fmtEl,qEl].forEach(el=>{
    el.addEventListener('input', ()=>{ if(img) scheduleEstimate(); });
    el.addEventListener('change', ()=>{ if(img) scheduleEstimate(); });
  });

  // Reset
  resetBtn.addEventListener('click', ()=>{
    if(imgURL){ URL.revokeObjectURL(imgURL); imgURL=null; }
    img=null; baseName='image'; rot=0; flip=1;
    view.cx=0; view.cy=0;
    modeEl.value='fill'; zoomEl.value=100; bgEl.value='#000000';
    guidesEl.value='on'; overlayOpacityEl.value=70;
    fmtEl.value='image/jpeg'; qEl.value=92;
    fileInput.value=''; cameraInput.value='';
    ctx.clearRect(0,0,pv.width,pv.height);
    meta.hidden=true; downloadFull.disabled = downloadDesktop.disabled = downloadSafe.disabled = true;
    draw();
  });

  // Init
  window.addEventListener('resize', draw);
  document.getElementById('vt-year').textContent = new Date().getFullYear();
  draw();
})();
