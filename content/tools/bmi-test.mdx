<script>
(function(){
  const unitSel = document.getElementById('unit');
  const metric = document.getElementById('metric-fields');
  const imperial = document.getElementById('imperial-fields');
  const calcBtn = document.getElementById('calc');
  const printBtn = document.getElementById('print');

  const ageEl = document.getElementById('age');
  const kgEl = document.getElementById('kg');
  const cmEl = document.getElementById('cm');
  const lbEl = document.getElementById('lb');
  const ftEl = document.getElementById('ft');
  const inchEl = document.getElementById('inch');
  const waistEl = document.getElementById('waist');

  const bmiLine = document.getElementById('bmi-line');
  const bmiPill = document.getElementById('bmi-pill');
  const seniorNote = document.getElementById('senior-note');
  const wthrBox = document.getElementById('wthr');

  const adWrap = document.getElementById('calc-ad-slot');

  unitSel.addEventListener('change', () => {
    if (unitSel.value === 'imperial'){
      metric.style.display = 'none';
      imperial.style.display = 'grid';
      waistEl.placeholder = "in";
    } else {
      metric.style.display = 'grid';
      imperial.style.display = 'none';
      waistEl.placeholder = "cm";
    }
  });

  function toMeters(){
    if(unitSel.value === 'metric'){
      const cm = parseFloat(cmEl.value);
      return (isFinite(cm) && cm > 0) ? (cm/100) : NaN;
    } else {
      const ft = parseFloat(ftEl.value) || 0;
      const inch = parseFloat(inchEl.value) || 0;
      const totalIn = ft*12 + inch;
      return (totalIn > 0) ? (totalIn * 0.0254) : NaN;
    }
  }
  function toKg(){
    if(unitSel.value === 'metric'){
      const kg = parseFloat(kgEl.value);
      return (isFinite(kg) && kg > 0) ? kg : NaN;
    } else {
      const lb = parseFloat(lbEl.value);
      return (isFinite(lb) && lb > 0) ? (lb * 0.45359237) : NaN;
    }
  }
  function waistToHeightRatio(){
    const h = toMeters();
    if(!isFinite(h)) return NaN;
    let waist = parseFloat(waistEl.value);
    if(!isFinite(waist) || waist <= 0) return NaN;
    if(unitSel.value === 'imperial'){ waist = waist * 2.54; } // inches -> cm
    const heightCm = h*100;
    return waist/heightCm;
  }

  function catFromBMI(b){
    if(b < 18.5) return {label:'Underweight', tone:'danger'};
    if(b < 25) return {label:'Healthy weight', tone:'ok'};
    if(b < 30) return {label:'Overweight', tone:'warn'};
    if(b < 35) return {label:'Obesity (Class I)', tone:'danger'};
    if(b < 40) return {label:'Obesity (Class II)', tone:'danger'};
    return {label:'Obesity (Class III)', tone:'danger'};
  }

  function setPill(text, tone){
    bmiPill.textContent = text;
    bmiPill.style.display = 'inline-flex';
    bmiPill.className = 'pill ' + (tone === 'ok' ? 'pill--ok' : tone === 'warn' ? 'pill--warn' : 'pill--danger');
  }

  function seniorContext(age, bmi){
    if(!isFinite(age)) return "BMI is a simple screening number. Discuss concerns with your clinician.";
    if(age >= 65){
      if(bmi < 20) return "In older adults, very low BMI can relate to frailty, falls, and slower recovery. If weight is dropping unintentionally, talk to your clinician.";
      if(bmi >= 20 && bmi < 23) return "For some older adults, a slightly higher BMI may be acceptable. Consider strength, mobility, and recent weight changes.";
      if(bmi >= 23 && bmi < 30) return "Context matters: function, strength, and waist size provide extra insight alongside BMI.";
      return "Consider discussing waist size, activity, and any limits with your clinician to tailor a safe plan.";
    }
    return "Use BMI together with waist size, strength, and how you feel day to day.";
  }

  function renderAdEachCalc(){
    // Make the ad container visible and request a fresh fill each time
    adWrap.style.display = 'block';
    try {
      (adsbygoogle = window.adsbygoogle || []).push({});
    } catch(e){}
  }

  calcBtn.addEventListener('click', () => {
    const m = toMeters();
    const kg = toKg();
    if(!isFinite(m) || !isFinite(kg)){
      bmiLine.textContent = "Your BMI: — (enter valid height and weight)";
      bmiPill.style.display = 'none';
      seniorNote.textContent = "";
      wthrBox.style.display = 'none';
      return;
    }
    const bmi = kg / (m*m);
    const bmiRounded = Math.round(bmi*10)/10;
    const cat = catFromBMI(bmi);
    bmiLine.textContent = `Your BMI: ${bmiRounded}`;
    setPill(cat.label, cat.tone);

    const age = parseFloat(ageEl.value);
    seniorNote.textContent = seniorContext(age, bmi);

    const r = waistToHeightRatio();
    if(isFinite(r)){
      const rRound = Math.round(r*100)/100;
      let msg = `Waist-to-Height Ratio (WHtR): ${rRound}`;
      if(r < 0.5) { msg += " — generally within a favorable range."; wthrBox.className="ok"; }
      else { msg += " — consider discussing with your clinician."; wthrBox.className="note"; }
      wthrBox.textContent = msg;
      wthrBox.style.display = 'block';
    } else {
      wthrBox.style.display = 'none';
    }

    renderAdEachCalc(); // show AdSense unit on each calculation
  });

  printBtn.addEventListener('click', () => {
    window.print();
  });

  // Small QoL: Esc to blur on mobile keyboards
  document.addEventListener('keydown', (e)=>{ if(e.key==='Escape') document.activeElement?.blur(); });
})();
