    c.beginPath();
    c.moveTo(x+r, y);
    c.arcTo(x+w, y, x+w, y+h, r);
    c.arcTo(x+w, y+h, x, y+h, r);
    c.arcTo(x, y+h, x, y, r);
    c.arcTo(x, y, x+w, y, r);
    c.closePath();
  }

        const blob = await canvasToBlobPromise(off, mime, mid/100);
        if(blob.size <= target){ best = blob; lo = mid + 1; }
        else { hi = mid - 1; }
      }
      return best || await canvasToBlobPromise(off, mime, (qualityEl.value|0)/100);
    }else{
      const q = (mime==='image/png') ? 1 : (qualityEl.value|0)/100;
      return await canvasToBlobPromise(off, mime, q);
    }
  }
  function canvasToBlobPromise(canvas, mime, q){
    return new Promise(res=> canvas.toBlob(b=> res(b), mime, q));
  }

  function loadFile(file){
    if(!file || !file.type?.startsWith('image/')) return;
    baseName = (file.name||'image').replace(/\.[^.]+$/,'');
    if(imgURL) URL.revokeObjectURL(imgURL);
    imgURL = URL.createObjectURL(file);
    const im = new Image();
    im.onload = ()=>{
      img = im; rot=0; flip=1; fileInput.value=''; cameraInput.value='';
      resetView(); draw(); downloadBtn.disabled=false;
    };
    im.onerror = ()=>{ URL.revokeObjectURL(imgURL); fileInput.value=''; cameraInput.value=''; alert('Could not load this image.'); };
    im.src = imgURL;
  }

  // Drag & drop + paste
  const drop = $('#drop');
  ['dragenter','dragover'].forEach(ev=> drop.addEventListener(ev, e=>{ e.preventDefault(); drop.classList.add('drag'); }));
  ['dragleave','drop'].forEach(ev=> drop.addEventListener(ev, e=>{ e.preventDefault(); drop.classList.remove('drag'); }));
  drop.addEventListener('drop', e=>{ const f=e.dataTransfer?.files?.[0]; if(f) loadFile(f); });
  window.addEventListener('paste', e=>{
    const items = e.clipboardData?.items||[];
    for(const it of items){ if(it.type?.startsWith('image/')){ loadFile(it.getAsFile()); break; } }
  });

  // Inputs
  fileInput.addEventListener('change', e=>{ const f=e.target.files?.[0]; if(f) loadFile(f); });
  cameraInput.addEventListener('change', e=>{ const f=e.target.files?.[0]; if(f) loadFile(f); });

  // Provider/doc changes
  providerEl.addEventListener('change', applyPreset);
  docTypeEl.addEventListener('change', applyPreset);
  [outWEl,outHEl,kbLimitEl,formatEl].forEach(el=>{
    el.addEventListener('input', ()=>{ updateSpec(); draw(); });
    el.addEventListener('change', ()=>{ updateSpec(); draw(); });
  });

  // Zoom + pan
  zoomEl.addEventListener('input', ()=>{
    view.scale = (+zoomEl.value)/100;
    draw();
  });

  function startDrag(e){
    if(!img) return;
    dragging = true;
    const pt = ('touches' in e) ? e.touches[0] : e;
    lastX = pt.clientX; lastY = pt.clientY;
    e.preventDefault?.();
  }
  function moveDrag(e){
    if(!dragging || !img) return;
    const pt = ('touches' in e) ? e.touches[0] : e;
    const dx = pt.clientX - lastX, dy = pt.clientY - lastY;
    lastX = pt.clientX; lastY = pt.clientY;

    // Convert preview dx,dy to image pixels (approx via crop scale)
    const CW = pv.width * 0.88;
    const scalePreview = (CW / (+outWEl.value));
    // Account for rotation (drag in preview space -> rotate back)
    const ang = -rot * Math.PI/180;
    const rx =  dx*Math.cos(ang) - dy*Math.sin(ang);
    const ry =  dx*Math.sin(ang) + dy*Math.cos(ang);
    view.cx -= rx / (view.scale * scalePreview) * (1/flip);
    view.cy -= ry / (view.scale * scalePreview);
    // clamp to image bounds a bit outside to allow edge fill
    view.cx = Math.max(-img.width, Math.min(2*img.width, view.cx));
    view.cy = Math.max(-img.height, Math.min(2*img.height, view.cy));
    draw();
  }
  function endDrag(){ dragging=false; }

  pv.addEventListener('mousedown', startDrag);
  window.addEventListener('mousemove', moveDrag);
  window.addEventListener('mouseup', endDrag);
  pv.addEventListener('touchstart', startDrag, {passive:false});
  window.addEventListener('touchmove', moveDrag, {passive:false});
  window.addEventListener('touchend', endDrag);

  // Signature toggles
  bwEl.addEventListener('change', ()=>{ thWrap.classList.toggle('hide', !bwEl.checked); draw(); });
  [thEl,bgEl].forEach(el=> el.addEventListener('input', draw));

  // Rotate/flip
  rotL.addEventListener('click', ()=>{ rot=(rot-90+360)%360; resetView(); draw(); });
  rotR.addEventListener('click', ()=>{ rot=(rot+90)%360; resetView(); draw(); });
  flipH.addEventListener('click', ()=>{ flip*=-1; draw(); });

  // Download
  downloadBtn.addEventListener('click', async ()=>{
    if(!img) return;
    const blob = await exportImage();
    if(!blob){ alert('Could not produce output.'); return; }
    const extMap={'image/jpeg':'jpg','image/png':'png','image/webp':'webp'};
    const ext=extMap[formatEl.value]||'jpg';
    const a=document.createElement('a');
    a.href=URL.createObjectURL(blob);
    const suffix = (docTypeEl.value==='photo') ? 'photo' : 'sign';
    a.download=`${baseName}-pan-${suffix}.${ext}`;
    document.body.appendChild(a); a.click();
    setTimeout(()=>{ URL.revokeObjectURL(a.href); a.remove(); }, 600);
  });

  // Reset
  resetBtn.addEventListener('click', ()=>{
    if(imgURL){ URL.revokeObjectURL(imgURL); imgURL=null; }
    img=null; rot=0; flip=1; baseName='image';
    providerEl.value='nsdl'; docTypeEl.value='photo';
    applyPreset();
    fileInput.value=''; cameraInput.value='';
    ctx.clearRect(0,0,pv.width,pv.height);
    downloadBtn.disabled=true;
  });

  // Init
  window.addEventListener('resize', draw);
  applyPreset();
  draw();
})();
