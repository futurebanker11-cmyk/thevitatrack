<script>
(()=> {
  const $ = s => document.querySelector(s);
  const list = $('#list');
  const drop = $('#drop');
  const fileInput = $('#file');
  const cameraInput = $('#cameraFile');
  const dirInput = $('#dirFile');

  const modeEl = $('#mode');
  const fitWrap = $('#fitWrap'), fitW = $('#fitW'), fitH = $('#fitH'), upscale = $('#upscale');
  const longWrap = $('#longWrap'), longEdge = $('#longEdge'), upscale2 = $('#upscale2');
  const percentWrap = $('#percentWrap'), percent = $('#percent');
  const canvasWrap = $('#canvasWrap'), canvasMore = $('#canvasMore'), canW = $('#canW'), canH = $('#canH'), canvasFit = $('#canvasFit'), bgColor = $('#bgColor'), bgAlpha = $('#bgAlpha');

  const formatEl = $('#format'), qEl = $('#quality'), renameEl = $('#rename');
  const processBtn = $('#processBtn'), resetBtn = $('#resetBtn');
  const progress = $('#progress');
  const stats = $('#stats'), countInfo = $('#countInfo'), estInfo = $('#estInfo'), dimInfo = $('#dimInfo');

  document.getElementById('vt-year').textContent = new Date().getFullYear();

  // State
  let items = []; // {file, url, name, ext, img, w, h}
  let processing = false;

  // ===== Helpers =====
  function extForMime(mime){ return mime==='image/png' ? 'png' : (mime==='image/webp' ? 'webp' : 'jpg'); }
  function pad(n, w){ const s=String(n); return s.length>=w?s:'0'.repeat(w-s.length)+s; }
  function humanBytes(bytes){
    if(!bytes) return '—';
    const u=['B','KB','MB','GB']; let i=0; let v=bytes;
    while(v>1024 && i<u.length-1){ v/=1024; i++; }
    return v.toFixed(v<10?2:1)+' '+u[i];
  }
  function hexWithAlpha(hex, a){
    const m = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
    if(!m) return `rgba(0,0,0,${a})`;
    const r = parseInt(m[1],16), g=parseInt(m[2],16), b=parseInt(m[3],16);
    return `rgba(${r},${g},${b},${a})`;
  }

  function toggleModeBlocks(){
    fitWrap.classList.toggle('hide', modeEl.value!=='fit');
    longWrap.classList.toggle('hide', modeEl.value!=='long');
    percentWrap.classList.toggle('hide', modeEl.value!=='percent');
    const isCanvas = (modeEl.value==='canvas');
    canvasWrap.classList.toggle('hide', !isCanvas);
    canvasMore.classList.toggle('hide', !isCanvas);
    updateAllOutDims();
  }

  function addFiles(fileList){
    const arr = Array.from(fileList || []);
    if(!arr.length) return;
    arr.forEach(file=>{
      if(!file.type?.startsWith('image/')) return;
      const url = URL.createObjectURL(file);
      const name = (file.name||'image').replace(/\.[^.]+$/,'');
      const ext = (file.name.match(/\.([^.]+)$/i)?.[1]||'').toLowerCase();
      const img = new Image();
      img.onload = ()=>{
        items.push({file,url,name,ext,img,w:img.naturalWidth,h:img.naturalHeight});
        renderList();
        updateStats();
        processBtn.disabled = items.length===0;
      };
      img.onerror = ()=> URL.revokeObjectURL(url);
      img.src = url;
    });
  }

  function removeItem(idx){
    const it = items[idx];
    if(it?.url) URL.revokeObjectURL(it.url);
    items.splice(idx,1);
    renderList();
    updateStats();
    processBtn.disabled = items.length===0;
  }

  function clearAll(){
    items.forEach(it=>{ if(it.url) URL.revokeObjectURL(it.url); });
    items = [];
    renderList();
    updateStats();
    processBtn.disabled = true;
  }

  function renderList(){
    list.innerHTML = '';
    items.forEach((it, idx)=>{
      const card = document.createElement('div'); card.className='item';
      const img = document.createElement('img'); img.className='thumb'; img.src = it.url; img.alt = it.name;
      const meta = document.createElement('div'); meta.className='meta';
      const nm = document.createElement('div'); nm.className='name'; nm.textContent = it.name + (it.ext?'.'+it.ext:'');
      const dims = document.createElement('div'); dims.className='dims'; dims.textContent = `${it.w}×${it.h}px`;
      const out = document.createElement('div'); out.className='out'; out.id = `out-${idx}`; out.textContent = '→ —';
      const rm = document.createElement('div'); rm.className='rm';
      const btn = document.createElement('button'); btn.type='button'; btn.textContent='Remove';
      btn.addEventListener('click', ()=> removeItem(idx));
      rm.appendChild(btn);
      meta.appendChild(nm); meta.appendChild(dims); meta.appendChild(out);
      card.appendChild(img); card.appendChild(meta); card.appendChild(rm);
      list.appendChild(card);
    });
    updateAllOutDims();
  }

  function currentDimsFor(w,h){
    const mode = modeEl.value;
    if(mode==='fit'){
      const W = Math.max(1, +fitW.value||1);
      const H = Math.max(1, +fitH.value||1);
      let s = Math.min(W/w, H/h);
      if(!upscale.checked) s = Math.min(1, s);
      return {tw: Math.max(1, Math.round(w*s)), th: Math.max(1, Math.round(h*s))};
    }else if(mode==='long'){
      const L = Math.max(1, +longEdge.value||1);
      const m = Math.max(w,h);
      let s = L / m;
      if(!upscale2.checked) s = Math.min(1, s);
      return {tw: Math.max(1, Math.round(w*s)), th: Math.max(1, Math.round(h*s))};
    }else if(mode==='percent'){
      const p = Math.max(1, Math.min(400, +percent.value||100));
      const s = p/100;
      return {tw: Math.max(1, Math.round(w*s)), th: Math.max(1, Math.round(h*s))};
    }else{ // canvas
      const W = Math.max(1, +canW.value||1);
      const H = Math.max(1, +canH.value||1);
      if(canvasFit.value==='contain'){
        const s = Math.min(W/w, H/h);
        return {tw: Math.max(1, Math.round(w*s)), th: Math.max(1, Math.round(h*s)), cw:W, ch:H };
      }else{
        // cover: final canvas W×H, draw scaled to fill
        const s = Math.max(W/w, H/h);
        return {tw: Math.round(w*s), th: Math.round(h*s), cw:W, ch:H};
      }
    }
  }

  function updateAllOutDims(){
    items.forEach((it, idx)=>{
      const d = currentDimsFor(it.w, it.h);
      const el = document.getElementById(`out-${idx}`);
      if(el){
        if(d.cw && d.ch){
          el.textContent = canvasFit.value==='contain'
            ? `→ ${d.cw}×${d.ch} (image ~ ${d.tw}×${d.th})`
            : `→ ${d.cw}×${d.ch} (cover from ${d.tw}×${d.th})`;
        }else{
          el.textContent = `→ ${d.tw}×${d.th}`;
        }
      }
    });
    updateStats();
  }

  function updateStats(){
    countInfo.textContent = `Files: ${items.length}`;
    let modeText = modeEl.options[modeEl.selectedIndex].textContent;
    if(modeEl.value==='fit') modeText += ` (${fitW.value}×${fitH.value}${upscale.checked? ', upscale':''})`;
    if(modeEl.value==='long') modeText += ` (${longEdge.value}${upscale2.checked? ', upscale':''})`;
    if(modeEl.value==='percent') modeText += ` (${percent.value}%)`;
    if(modeEl.value==='canvas') modeText += ` (${canW.value}×${canH.value}, ${canvasFit.value})`;
    dimInfo.textContent = `Mode: ${modeText}`;
    stats.hidden = items.length===0;
    estInfo.textContent = `Estimated ZIP: ${items.length? 'calculating…' : '—'}`;
    // Quick client-side estimate: assume similar size scaling as pixel count scaling (very rough)
    if(items.length){
      const fmt = formatEl.value;
      const q = +qEl.value||90;
      let totalPixelsSrc = 0, totalPixelsDst = 0;
      items.forEach(it=>{
        totalPixelsSrc += it.w*it.h;
        const d = currentDimsFor(it.w,it.h);
        const pw = d.cw? d.cw : d.tw;
        const ph = d.ch? d.ch : d.th;
        totalPixelsDst += pw*ph;
      });
      const ratio = totalPixelsDst/Math.max(1,totalPixelsSrc);
      // assume baseline 75% of source total bytes for photos at q≈90, crude
      const approx = items.reduce((s,it)=> s + (it.file.size * ratio * (fmt==='image/png'?1.1:(fmt==='image/webp'?0.6:0.75))), 0);
      // add zip overhead ~ 80 bytes per file
      const approxZip = approx + items.length*80;
      estInfo.textContent = `Estimated ZIP: ${humanBytes(approxZip)}`;
    }
  }

  // ===== Resizing & Export =====
  function drawResized(img, d){
    const mime = formatEl.value;
    const q = (mime==='image/png') ? 1 : (+qEl.value||90)/100;

    let canvas = document.createElement('canvas');
    let ctx = canvas.getContext('2d');
    ctx.imageSmoothingEnabled = true; ctx.imageSmoothingQuality='high';

    if(d.cw && d.ch){
      // Canvas modes
      canvas.width = d.cw; canvas.height = d.ch;
      ctx.clearRect(0,0,canvas.width,canvas.height);
      if(canvasFit.value==='contain'){
        // background
        const a = (+bgAlpha.value||0)/100;
        ctx.fillStyle = hexWithAlpha(bgColor.value, a);
        ctx.fillRect(0,0,canvas.width,canvas.height);
        // draw centered
        const x = Math.round((canvas.width - d.tw)/2);
        const y = Math.round((canvas.height - d.th)/2);
        ctx.drawImage(img, 0,0, img.naturalWidth, img.naturalHeight, x,y, d.tw,d.th);
      }else{
        // cover: draw scaled, centered, cropping overflow
        const tw = d.tw, th = d.th; // scaled size
        const x = Math.round((canvas.width - tw)/2);
        const y = Math.round((canvas.height - th)/2);
        ctx.drawImage(img, 0,0, img.naturalWidth, img.naturalHeight, x,y, tw,th);
      }
    }else{
      canvas.width = d.tw; canvas.height = d.th;
      ctx.drawImage(img, 0,0, img.naturalWidth, img.naturalHeight, 0,0, d.tw, d.th);
    }
    return new Promise(res=> canvas.toBlob(b=> res(b), mime, q));
  }

  function makeName(base, i, w, h){
    const ext = extForMime(formatEl.value);
    let name = renameEl.value || '{name}-{w}x{h}-{i}';
    name = name.replace(/{name}/g, base)
               .replace(/{w}/g, w)
               .replace(/{h}/g, h)
               .replace(/{i}/g, pad(i,2));
    return `${name}.${ext}`;
  }

  // Minimal ZIP (STORE, no compression)
  // CRC32
  const crcTable = (()=> {
    let c; const table = new Uint32Array(256);
    for(let n=0;n<256;n++){
      c=n; for(let k=0;k<8;k++){ c = (c & 1) ? (0xEDB88320 ^ (c>>>1)) : (c>>>1); }
      table[n]=c>>>0;
    } return table;
  })();
  function crc32Uint8(u8){
    let c = 0 ^ (-1);
    for(let i=0;i<u8.length;i++){ c = (c>>>8) ^ crcTable[(c ^ u8[i]) & 0xFF]; }
    return (c ^ (-1)) >>> 0;
  }
  function u16(v){ const b=new Uint8Array(2); b[0]=v&255; b[1]=(v>>>8)&255; return b; }
  function u32(v){ const b=new Uint8Array(4); b[0]=v&255; b[1]=(v>>>8)&255; b[2]=(v>>>16)&255; b[3]=(v>>>24)&255; return b; }
  function dosTimeDate(d){
    const dt = d instanceof Date ? d : new Date();
    const dosTime = ((dt.getHours()&31)<<11) | ((dt.getMinutes()&63)<<5) | ((Math.floor(dt.getSeconds()/2))&31);
    const dosDate = (((dt.getFullYear()-1980)&127)<<9) | (((dt.getMonth()+1)&15)<<5) | (dt.getDate()&31);
    return {dosTime, dosDate};
  }
  async function blobToUint8(blob){ return new Uint8Array(await blob.arrayBuffer()); }
  function strU8(s){
    return new TextEncoder().encode(s);
  }
  async function makeZip(entries){
    // entries: [{name, data:Uint8Array}]
    const chunks = [];
    const central = [];
    let offset = 0;
    const now = dosTimeDate(new Date());
    for(const e of entries){
      const nameU8 = strU8(e.name);
      const data = e.data;
      const crc = crc32Uint8(data);
      const compSize = data.length, uncompSize = data.length;

      // Local file header
      chunks.push(
        u32(0x04034b50),     // local header sig
        u16(20),             // version needed
        u16(0),              // flags
        u16(0),              // method 0=store
        u16(now.dosTime),    // time
        u16(now.dosDate),    // date
        u32(crc), u32(compSize), u32(uncompSize),
        u16(nameU8.length),  // fname len
        u16(0),              // extra len
        nameU8,              // fname
        data                 // file data
      );
      const localLen = 30 + nameU8.length + data.length;

      // Central directory header (gathers)
      central.push({
        header: [
          u32(0x02014b50),   // central sig
          u16(20),           // version made by
          u16(20),           // version needed
          u16(0),            // flags
          u16(0),            // method
          u16(now.dosTime), u16(now.dosDate),
          u32(crc), u32(compSize), u32(uncompSize),
          u16(nameU8.length), u16(0), u16(0), // fname, extra, comment
          u16(0), u16(0),    // disk start, internal attrs
          u32(0),            // external attrs
          u32(offset),       // relative offset of local
          nameU8
        ],
        size: 46 + nameU8.length
      });
      offset += localLen;
    }
    const startOfCentral = offset;
    central.forEach(c => { chunks.push(...c.header); offset += c.size; });
    const centralSize = offset - startOfCentral;

    // End of central directory
    chunks.push(
      u32(0x06054b50),
      u16(0), u16(0), // disk numbers
      u16(entries.length), u16(entries.length),
      u32(centralSize),
      u32(startOfCentral),
      u16(0) // comment len
    );

    // Flatten
    const flat = [];
    chunks.forEach(ch=>{
      if(ch instanceof Uint8Array) flat.push(ch);
      else if(Array.isArray(ch)) ch.forEach(x=> flat.push(x));
      else flat.push(ch);
    });
    return new Blob(flat, {type:'application/zip'});
  }
