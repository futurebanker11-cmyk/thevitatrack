<script>
(()=> {
  const $ = s => document.querySelector(s);

  // Elements
  const pv = $('#pv'), ctx = pv.getContext('2d');
  const wrap = $('#canvasWrap'), drop = $('#drop');
  const fileInput = $('#file'), cameraInput = $('#cameraFile');

  const presetEl = $('#preset');
  const rotL = $('#rotL'), rotR = $('#rotR');
  const fmtEl = $('#format'), qEl = $('#quality');

  const brightnessEl = $('#brightness'), contrastEl = $('#contrast'), gammaEl = $('#gamma');
  const highlightsEl = $('#highlights'), shadowsEl = $('#shadows');
  const vignetteEl = $('#vignette'), vigSoftEl = $('#vigSoft'), grainEl = $('#grain');

  const downloadBtn = $('#downloadBtn'), resetBtn = $('#resetBtn');

  const meta = $('#meta'), origInfo = $('#origInfo'), expInfo = $('#expInfo'), estInfo = $('#estInfo');

  // State
  let img = null, imgURL = null, baseName = 'image';
  let rot = 0; // 0/90/180/270
  let exportBlob = null;

  // Helpers
  function fitCanvas(){
    const w = Math.max(320, wrap.clientWidth - 24);
    const h = Math.max(260, Math.round(w*0.66));
    pv.width = w; pv.height = h;
  }
  function getRotDims(w,h,r){ return (r%180===0) ? {w,h} : {w:h, h:w}; }
  function humanBytes(bytes){
    if(!bytes) return '—';
    const u=['B','KB','MB','GB']; let i=0; let v=bytes;
    while(v>1024 && i<u.length-1){ v/=1024; i++; }
    return v.toFixed(v<10?2:1)+' '+u[i];
  }
  const clamp = (v,min,max)=>Math.max(min,Math.min(max,v));

  // ===== Filters pipeline =====
  function applyPresetValues(){
    const p = presetEl.value;
    if(p==='neutral'){ brightnessEl.value=0; contrastEl.value=20; gammaEl.value=100; highlightsEl.value=0; shadowsEl.value=10; vignetteEl.value=20; vigSoftEl.value=60; grainEl.value=8; }
    if(p==='high'){ brightnessEl.value=0; contrastEl.value=45; gammaEl.value=105; highlightsEl.value=-10; shadowsEl.value=20; vignetteEl.value=30; vigSoftEl.value=55; grainEl.value=12; }
    if(p==='matte'){ brightnessEl.value=4; contrastEl.value=-10; gammaEl.value=90; highlightsEl.value=-10; shadowsEl.value=35; vignetteEl.value=18; vigSoftEl.value=70; grainEl.value=10; }
    if(p==='film'){ brightnessEl.value=0; contrastEl.value=25; gammaEl.value=95; highlightsEl.value=-6; shadowsEl.value=18; vignetteEl.value=24; vigSoftEl.value=65; grainEl.value=24; }
    if(p==='sepia'){ brightnessEl.value=2; contrastEl.value=10; gammaEl.value=100; highlightsEl.value=0; shadowsEl.value=6; vignetteEl.value=18; vigSoftEl.value=70; grainEl.value=10; }
    if(p==='cyan'){ brightnessEl.value=0; contrastEl.value=18; gammaEl.value=105; highlightsEl.value=0; shadowsEl.value=10; vignetteEl.value=22; vigSoftEl.value=65; grainEl.value=12; }
    if(p==='news'){ brightnessEl.value=6; contrastEl.value=60; gammaEl.value=110; highlightsEl.value=-12; shadowsEl.value=30; vignetteEl.value=10; vigSoftEl.value=65; grainEl.value=0; }
  }

  function drawPreview(){
    fitCanvas();
    ctx.clearRect(0,0,pv.width,pv.height);
    if(!img){ meta.hidden = true; downloadBtn.disabled = true; return; }

    // Draw oriented, fit to canvas
    const {w:rw, h:rh} = getRotDims(img.width, img.height, rot);
    const scale = Math.min(pv.width/rw, pv.height/rh);
    const drawW = Math.max(1, Math.round(rw*scale));
    const drawH = Math.max(1, Math.round(rh*scale));

    ctx.save();
    ctx.imageSmoothingEnabled = true;
    ctx.imageSmoothingQuality = 'high';
    ctx.translate(pv.width/2, pv.height/2);
    ctx.rotate(rot*Math.PI/180);
    ctx.drawImage(img, -drawW/2, -drawH/2, drawW, drawH);
    ctx.restore();

    // Apply filters to the current pixels (skip fully transparent ones)
    applyBWOnContext(ctx, pv.width, pv.height, {
      tone: presetEl.value, // for tinting
      brightness: +brightnessEl.value || 0,
      contrast: +contrastEl.value || 0,
      gamma: (+gammaEl.value || 100)/100,
      highlights: +highlightsEl.value || 0,
      shadows: +shadowsEl.value || 0,
      vignette: +vignetteEl.value || 0,
      vigSoft: +vigSoftEl.value || 60,
      grain: +grainEl.value || 0
    });

    origInfo.textContent = `Original: ${img.width}×${img.height}px`;
    const {w:ew, h:eh} = getRotDims(img.width, img.height, rot);
    expInfo.textContent = `Export: ${ew}×${eh}px`;
    meta.hidden = false;

    scheduleEstimate();
    downloadBtn.disabled = false;
  }

  function applyBWOnContext(c2d, w, h, params){
    const id = c2d.getImageData(0,0,w,h);
    const d = id.data;
    const br = params.brightness; // -100..100
    const cr = params.contrast;   // -100..100
    const gm = clamp(params.gamma, 0.2, 5.0); // ~0.5..2.0
    const hi = params.highlights; // -100..100
    const sh = params.shadows;    // -100..100
    const vg = params.vignette;   // 0..100
    const vs = clamp(params.vigSoft/100, 0.1, 1.0);
    const gr = params.grain;      // 0..100
    const tone = params.tone;     // for sepia/cyan tinting

    // Precompute
    const cFactor = (259 * (cr + 255)) / (255 * (259 - cr)); // classic contrast around mid 128
    const brAdd = br * 1.28; // scale to ~[-128..128]
    const cx = w/2, cy = h/2;
    const maxDist = Math.hypot(cx, cy);

    // Grain amplitude in 0..~30
    const grainAmp = gr * 0.3;

    for(let y=0; y<h; y++){
      for(let x=0; x<w; x++){
        const i = (y*w + x) * 4;
        const a = d[i+3];
        if(a===0) continue; // ignore transparent bg

        let r=d[i], g=d[i+1], b=d[i+2];

        // Luminance (Rec. 709)
        let lum = 0.2126*r + 0.7152*g + 0.0722*b;

        // Tone tinting (after grayscale)
        let tr=lum, tg=lum, tb=lum;
        if(tone==='sepia'){
          tr = lum * 1.07; tg = lum * 0.85; tb = lum * 0.68;
        }else if(tone==='cyan'){
          tr = lum * 0.75; tg = lum * 0.90; tb = lum * 1.15;
        }else{
          tr = tg = tb = lum;
        }

        // Brightness / Contrast
        tr = cFactor*(tr-128)+128 + brAdd;
        tg = cFactor*(tg-128)+128 + brAdd;
        tb = cFactor*(tb-128)+128 + brAdd;

        // Highlights / Shadows (gentle curve toward mid 128)
        const hlK = hi/100, shK = sh/100;
        const adjHS = (v)=>{
          // normalized 0..1
          let vn = clamp(v/255, 0, 1);
          // shadows: lift low tones (when shK>0), or deepen (shK<0)
          vn = vn + shK * (0.5 - vn) * (1 - vn);
          // highlights: reduce high tones (hi>0), or boost (hi<0)
          vn = vn - hlK * (vn - 0.5) * (vn);
          return clamp(vn*255, 0, 255);
        };
        tr = adjHS(tr); tg = adjHS(tg); tb = adjHS(tb);

        // Gamma (mid-tone)
        const gCorr = (v)=> 255 * Math.pow(clamp(v/255,0,1), 1/gm);
        tr = gCorr(tr); tg = gCorr(tg); tb = gCorr(tb);

        // Vignette (darken edges)
        if(vg>0){
          const dist = Math.hypot(x-cx, y-cy);
          let vstr = (dist / maxDist);
          vstr = Math.pow(vstr, clamp(3*vs, 1.2, 4.0)); // softness
          const vMul = 1 - (vg/100) * vstr * 0.9;
          tr *= vMul; tg *= vMul; tb *= vMul;
        }

        // Grain (add noise)
        if(gr>0){
          const n = (Math.random()*2 - 1) * grainAmp;
          tr += n; tg += n; tb += n;
        }

        // Matte preset: add light lift in blacks / gentle roll-off highs
        if(tone==='matte'){
          const matte = (v)=>{
            const vn = clamp(v/255, 0, 1);
            const out = vn*0.9 + 0.05 + 0.05*Math.tanh((vn-0.5)*2.0);
            return clamp(out*255, 0, 255);
          };
          tr = matte(tr); tg = matte(tg); tb = matte(tb);
        }

        // Newspaper: hard threshold-ish with slight dithery noise
        if(tone==='news'){
          const n = (Math.random()-0.5)*18;
          const thr = 128 + n;
          const v = (tr+tg+tb)/3;
          const bw = v > thr ? 255 : 0;
          tr=tg=tb=bw;
        }

        d[i]   = clamp(tr,0,255)|0;
        d[i+1] = clamp(tg,0,255)|0;
        d[i+2] = clamp(tb,0,255)|0;
        // alpha unchanged
      }
    }
    c2d.putImageData(id, 0, 0);
  }

  // Export (full-size, oriented, then apply same pipeline)
  function exportImage(callback){
    if(!img){ callback(null); return; }
    const {w, h} = getRotDims(img.width, img.height, rot);
    const off = document.createElement('canvas');
    off.width = w; off.height = h;
    const octx = off.getContext('2d');
    octx.imageSmoothingEnabled = true; octx.imageSmoothingQuality = 'high';
    octx.translate(w/2, h/2);
    octx.rotate(rot*Math.PI/180);
    octx.drawImage(img, -img.width/2, -img.height/2);

    applyBWOnContext(octx, w, h, {
      tone: presetEl.value,
      brightness: +brightnessEl.value || 0,
      contrast: +contrastEl.value || 0,
      gamma: (+gammaEl.value || 100)/100,
      highlights: +highlightsEl.value || 0,
      shadows: +shadowsEl.value || 0,
      vignette: +vignetteEl.value || 0,
      vigSoft: +vigSoftEl.value || 60,
      grain: +grainEl.value || 0
    });

    const mime = fmtEl.value;
    const q = (mime==='image/png') ? 1 : (+qEl.value||92)/100;
    off.toBlob(b=> callback(b), mime, q);
  }

  // Size estimate
  let estRAF = null;
  function scheduleEstimate(){
    if(estRAF) cancelAnimationFrame(estRAF);
    estRAF = requestAnimationFrame(()=>{
      exportImage(b=>{
        exportBlob = b;
        estInfo.textContent = 'Estimated file size: ' + humanBytes(b?.size||0);
      });
    });
  }

  // File handling
  function loadFile(file){
    if(!file || !file.type?.startsWith('image/')) return;
    baseName = (file.name||'image').replace(/\.[^.]+$/,'');
    if(imgURL) URL.revokeObjectURL(imgURL);
    imgURL = URL.createObjectURL(file);
    const im = new Image();
    im.onload = ()=>{ img = im; fileInput.value=''; cameraInput.value=''; applyPresetValues(); drawPreview(); };
    im.onerror = ()=>{ URL.revokeObjectURL(imgURL); fileInput.value=''; cameraInput.value=''; alert('Could not load this image.'); };
    im.src = imgURL;
  }

  // Drag & drop
  ['dragenter','dragover'].forEach(ev=>{
    drop.addEventListener(ev, e=>{ e.preventDefault(); drop.classList.add('drag'); });
  });
  ['dragleave','drop'].forEach(ev=>{
    drop.addEventListener(ev, e=>{ e.preventDefault(); drop.classList.remove('drag'); });
  });
  drop.addEventListener('drop', e=>{
    const f = e.dataTransfer?.files?.[0];
    if(f) loadFile(f);
  });

  // Real inputs (mobile-safe)
  fileInput.addEventListener('change', e=>{
    const f=e.target.files && e.target.files[0];
    if(f) loadFile(f);
  });
  cameraInput.addEventListener('change', e=>{
    const f=e.target.files && e.target.files[0];
    if(f) loadFile(f);
  });

  // Paste
  window.addEventListener('paste', e=>{
    const items = e.clipboardData?.items;
    if(!items) return;
    for(const it of items){
      if(it.type?.startsWith('image/')){ loadFile(it.getAsFile()); break; }
    }
  });

  // Controls
  presetEl.addEventListener('change', ()=>{ applyPresetValues(); drawPreview(); });

  [brightnessEl, contrastEl, gammaEl, highlightsEl, shadowsEl, vignetteEl, vigSoftEl, grainEl]
    .forEach(el=>{
      el.addEventListener('input', drawPreview);
      el.addEventListener('change', drawPreview);
    });

  rotL.addEventListener('click', ()=>{ rot = (rot - 90 + 360) % 360; drawPreview(); });
  rotR.addEventListener('click', ()=>{ rot = (rot + 90) % 360; drawPreview(); });

  [fmtEl, qEl].forEach(el=>{
    el.addEventListener('input', ()=>{ if(img) scheduleEstimate(); });
    el.addEventListener('change', ()=>{ if(img) scheduleEstimate(); });
  });

  // Download
  downloadBtn.addEventListener('click', ()=>{
    if(!img) return;
    exportImage(blob=>{
      if(!blob) return;
      const extMap = {'image/jpeg':'jpg','image/png':'png','image/webp':'webp'};
      const ext = extMap[fmtEl.value]||'jpg';
      const a = document.createElement('a');
      a.href = URL.createObjectURL(blob);
      a.download = `${baseName}-bw.${ext}`;
      document.body.appendChild(a);
      a.click();
      setTimeout(()=>{ URL.revokeObjectURL(a.href); a.remove(); }, 500);
    });
  });

  // Reset
  resetBtn.addEventListener('click', ()=>{
    if(imgURL){ URL.revokeObjectURL(imgURL); imgURL=null; }
    img=null; baseName='image'; rot=0;
    presetEl.value='neutral'; applyPresetValues();
    fmtEl.value='image/jpeg'; qEl.value=92;
    fileInput.value=''; cameraInput.value='';
    ctx.clearRect(0,0,pv.width,pv.height);
    exportBlob=null; meta.hidden=true; downloadBtn.disabled=true;
  });

  // Init
  window.addEventListener('resize', drawPreview);
  document.getElementById('vt-year').textContent = new Date().getFullYear();
  applyPresetValues();
  drawPreview();
})();
</script>
